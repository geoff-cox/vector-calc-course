\begin{frame}
\frametitle{I. Introduction}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

The next test we will discuss is called the integral test.  This test relies on the fact that: \\

\begin{center}
	$\D{\sum \limits_{i=1}^{\infty} a_i}$  \qquad has a strong connection to \qquad \ON<2->{\B{$\D{\int \limits_{1}^{\infty} a(x) \, dx}$}}
\end{center}

The main idea is that the convergence of the series can be determined by the convergence of the improper integral. \\

\vspace{0.5cm}

For example, the convergence of \\
\vspace{0.2cm}
%-------NUM LIST---------
\begin{enumerate}
	\item[$\circ$] \quad $\D{\sum \limits_{k=1}^{\infty} \frac{\ln k}{k}}$ \quad can be determined by the convergence of \quad 
	\ON<2->{\B{$\D{\int \limits_{1}^{\infty} \frac{\ln x}{x} \, dx}$}}
	\item[$\circ$] \quad $\D{\sum \limits_{k=1}^{\infty} \frac{1}{k\ln k}}$ \quad can be determined by the convergence of \quad 
	\ON<2->{\B{$\D{\int \limits_{1}^{\infty} \frac{1}{x\ln x} \, dx}$}}
	\item[$\circ$] \quad $\D{\sum \limits_{k=1}^{\infty} \frac{\ln k}{k^2}}$ \quad can be determined by the convergence of \quad 
	\ON<2->{\B{$\D{\int \limits_{1}^{\infty} \frac{\ln x}{x^2} \, dx}$}}
\end{enumerate}
%------------------------

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{II. The Integral Test}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Suppose $f(x)$ is a function such that $f(n) = a_n$ .\quad $\left(\text{e.g. if} \quad \D{a_n = \frac{n}{n-1}} \quad \text{then} \quad \D{f(x) = \frac{x}{x-1}}\right)$

\vspace{0.5cm}

\begin{columns}\fcolorbox{blue}{white}{\begin{column}{10.4cm}
% Type the contents of the box here
{\bf Integral Test}:\\
\vspace{0.2cm}

If $f(x)$ is 
%-------LIST---------
\begin{itemize}
	\item[1.] \ON<2->{\B{positive on $[1,\infty)$\quad (or ``eventually'' positive)}}
	\vspace{0.2cm}
	\item[2.] \ON<3->{\B{continuous on $[1,\infty)$\quad (or ``eventually'' continuous)}}
	\vspace{0.2cm}
	\item[3.] \ON<4->{\B{decreasing on $[1,\infty)$\quad (or ``eventually'' decreasing)}}
\end{itemize}
%------------------- 
Then
\begin{center}
$\D{\int \limits_{1}^{\infty} f(x) \, dx \quad \text{converges} \quad \implies \quad  \sum \limits_{n=1}^{\infty}\, a_n}$ \quad \text{\ON<5->{\B{converges}}}\\
or\\
$\D{\int \limits_{1}^{\infty} f(x) \, dx \quad \text{diverges} \quad \implies \quad  \sum \limits_{n=1}^{\infty}\, a_n}$ \quad \text{\ON<6->{\B{diverges}}}\\
\vspace{0.2cm}
\end{center}
\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{columns}
\begin{column}{12.0cm}
	Use the integral test to determine the convergence of \quad $\D{\sum \limits_{n=1}^{\infty} \frac{1}{n}}$
	
%\vspace{0.2cm}
%
%\begin{tabular}{| c | c | p{7.6cm} |}

%		\hline
%	\end{tabular}

\vspace{0.2cm}

\underline{Step 1}: \quad Let 
	\ON<2->{\B{$f(x) = \frac{1}{x}$ on $[1,\infty)$}}\\
\vspace{0.5cm}
\underline{Step 2}: \quad (i) $f$ is positive since 
	\ON<3->{\B{$x$ is positive on $[1,\infty)$}}\\
\vspace{0.2cm}
\hspace{0.9cm}\quad (ii) \ON<4->{\B{$\D{\lim \limits_{x \to \infty} \, \frac{1}{x} = 0}$}}\\
\vspace{0.5cm}
\underline{Step 3}: \quad $\D{\int \limits_{1}^{\infty} \frac{1}{x} \, dx = 
	\ON<6->{\B{\lim \limits_{R \to \infty}\, \int \limits_{1}^{R} \frac{1}{x} \, dx}} 
	\ON<7->{\B{= \lim \limits_{R \to \infty}\, \left. \ln x \right|_1^R}} 
	\ON<8->{\B{= \lim \limits_{R \to \infty}\, \ln(R) - \ln(1) = \infty - 0 = \infty}}}$\\
\vspace{0.5cm}
\underline{Step 4}: \quad Since \quad $\D{\int \limits_{1}^{\infty} \frac{1}{x} \, dx}$ 
	\ON<9->{\B{\quad diverges \quad}} $\implies$ $\D{\quad \sum \limits_{n=1}^{\infty}\, a_n}$ 
	\ON<10->{\B{\quad diverges}}
