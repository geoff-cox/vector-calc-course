\begin{frame}
\frametitle{II. The Integral Test}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Suppose $f(x)$ is a function such that $f(n) = a_n$ .\quad $\left(\text{e.g. if} \quad \D{a_n = \frac{n}{n-1}} \quad \text{then} \quad \D{f(x) = \frac{x}{x-1}}\right)$

\vspace{0.5cm}

\begin{columns}\fcolorbox{blue}{white}{\begin{column}{10.4cm}
% Type the contents of the box here
{\bf Integral Test}:\\
\vspace{0.2cm}

If $f(x)$ is 
%-------LIST---------
\begin{itemize}
	\item[1.] \onslide<2->{\textcolor{blue}{positive on $[1,\infty)$\quad (or ``eventually'' positive)}}
	\vspace{0.2cm}
	\item[2.] \onslide<3->{\textcolor{blue}{continuous on $[1,\infty)$\quad (or ``eventually'' continuous)}}
	\vspace{0.2cm}
	\item[3.] \onslide<4->{\textcolor{blue}{decreasing on $[1,\infty)$\quad (or ``eventually'' decreasing)}}
\end{itemize}
%------------------- 
Then
\begin{center}
$\D{\int \limits_{1}^{\infty} f(x) \, dx \quad \text{converges} \quad \implies \quad  \sum \limits_{n=1}^{\infty}\, a_n}$ \quad \text{\onslide<5->{\textcolor{blue}{converges}}}\\
or\\
$\D{\int \limits_{1}^{\infty} f(x) \, dx \quad \text{diverges} \quad \implies \quad  \sum \limits_{n=1}^{\infty}\, a_n}$ \quad \text{\onslide<6->{\textcolor{blue}{diverges}}}\\
\vspace{0.2cm}
\end{center}
\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

{\bf Integral Test for $\sum \limits_{n=1}^{\infty}\, a_n$}: \quad Let $f(x)$ be a function such that $f(n)=a_n$. \quad If 
%-------LIST---------
\begin{itemize}
	\item[1.] \onslide<1->{\textcolor{blue}{positive on $[1,\infty)$\quad (or ``eventually'' positive)}}
	\vspace{0.2cm}
	\item[2.] \onslide<1->{\textcolor{blue}{continuous on $[1,\infty)$\quad (or ``eventually'' continuous)}}
	\vspace{0.2cm}
	\item[3.] \onslide<1->{\textcolor{blue}{decreasing on $[1,\infty)$\quad (or ``eventually'' decreasing)}}
\end{itemize}
%------------------- 
Then\\
\vspace{0.2cm}
$\D{\int \limits_{1}^{\infty} f(x) \, dx \quad \text{converges} \quad \implies \quad  \sum \limits_{n=1}^{\infty}\, a_n}$ \quad \text{\onslide<2->{\textcolor{blue}{converges}}} \qquad or\\
\vspace{0.1cm}
$\D{\int \limits_{1}^{\infty} f(x) \, dx \quad \text{diverges} \quad \implies \quad  \sum \limits_{n=1}^{\infty}\, a_n}$ \quad \text{\onslide<3->{\textcolor{blue}{diverges}}}\\
\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}

		Step 1 & {\bf C}onvert & $a_n \to f(x)$ defined on $[N,\infty)$ \quad ($N$ is some finite \#) \\

\vspace{0.2cm}

\underline{Step 1}: Let \onslide<2->{\textcolor{blue}{$f(x) = \frac{1}{x}$ on $[1,\infty)$}}\\
\vspace{0.3cm}
\underline{Step 2}: (i) $f$ is positive since \onslide<3->{\textcolor{blue}{$x$ is positive on $[1,\infty)$}}\\
\vspace{0.2cm}
\hspace{0.9cm}(ii) $f$ is strictly decreasing on $[1,\infty)$ since \onslide<4->{\textcolor{blue}{$x$ is strictly increasing on $[1,\infty)$}}\\
\vspace{0.2cm}
\hspace{0.9cm}(iii) $f$ is continuous on $[1,\infty)$ since \onslide<5->{\textcolor{blue}{$\frac{1}{x}$ is only discontinuous at $x=0$}}\\
\vspace{0.3cm}
\underline{Step 3}: $\D{\int \limits_{1}^{\infty} \frac{1}{x} \, dx = \onslide<6->{\textcolor{blue}{\lim \limits_{R \to \infty}\, \int \limits_{1}^{R} \frac{1}{x} \, dx}} \onslide<7->{\textcolor{blue}{= \lim \limits_{R \to \infty}\, \left. \ln x \right|_1^R}} \onslide<8->{\textcolor{blue}{= \lim \limits_{R \to \infty}\, \ln(R) - \ln(1) = \infty - 0 = \infty}}}$\\
\vspace{0.2cm}
\underline{Step 4}: Since $\D{\int \limits_{1}^{\infty} \frac{1}{x} \, dx}$ \onslide<9->{\textcolor{blue}{diverges}} $\implies$ $\D{\sum \limits_{n=1}^{\infty}\, a_n}$ \onslide<10->{\textcolor{blue}{diverges}}

\frametitle{I. P-Series Test}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Another useful test which is a result of the Integral Test is the {\bf P-Series Test} and is used to determine the convergence or divergence of series of the form

\vspace{0.4cm}

	\begin{columns}\fcolorbox{blue}{white}{\begin{column}{9.4cm}
% Type the contents of the box here
{\bf P-Series}:\\
\begin{center}
	$\D{\sum \limits_{n=1}^{\infty} \frac{1}{n^p}}$ = 
	$\left\{
	\begin{array}{l}
		\text{converges \,\, if \,\,} \onslide<2->{\textcolor{blue}{p > 1}} \\
		\text{diverges \,\, if \,\,} \onslide<3->{\textcolor{blue}{p \le 1}} \\
	\end{array}
	\right.$	
\end{center}

\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}

\vspace{0.4cm}

Determine if the following series converge or diverge: \\

\centering
%--------TABLE---------
	\setlength{\extrarowheight}{0.4cm}
	\begin{tabular}{c l l}
		1. & ${\sum \limits_{n=1}^{\infty} \frac{1}{n^{10}}}$ & \onslide<4->{\textcolor{blue}{converges since $p = 10 > 1$}}\\
		2. & ${\sum \limits_{n=2}^{\infty} n^{-1/2} \onslide<5->{\textcolor{blue}{= \sum \limits_{n=2}^{\infty} \frac{1}{n^{1/2}}}}}$ & \onslide<6->{\textcolor{blue}{diverges since $p = 1/2 \le 1$}}\\
		3. & ${\sum \limits_{n=1}^{\infty} \frac{1}{\sqrt[5]{n}} \onslide<7->{\textcolor{blue}{= \sum \limits_{n=1}^{\infty} \frac{1}{n^{1/5}}}}}$ & \onslide<8->{\textcolor{blue}{diverges since $p = 1/5 \le 1$}}\\
		4. & ${\sum \limits_{n=101}^{\infty} \frac{1}{n^{\pi}}}$ & \onslide<9->{\textcolor{blue}{converges since $p = \pi > 1$}}\\
		5. & ${\sum \limits_{n=1}^{\infty} \frac{\sqrt{2}n^{1/2}}{n^{3/2}} \onslide<10->{\textcolor{blue}{= \sum \limits_{n=1}^{\infty} \frac{\sqrt{2}}{n}}} \onslide<11->{\textcolor{blue}{ = \sqrt{2} \sum \limits_{n=1}^{\infty} \frac{1}{n}}}}$ & \onslide<12->{\textcolor{blue}{diverges since $p = 1 \le 1$}}\\
	\end{tabular}
