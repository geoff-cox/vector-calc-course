\begin{frame}
\frametitle{I. Sequences}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

A \underline{sequence}, $\{a_n\}$, is a \ON<2->{\B{finite or infinite list of terms}} separated by \ON<3->{\B{commas}}. \\

\vspace{0.2cm}

Ex: 
\begin{center}
	$\{a_n\} = \{2, 4,  8, 16, 32, ...\}$ \\
\end{center}
Each term is associated with an \ON<4->{\B{index.}}
\begin{center}
	\setlength{\extrarowheight}{0.2cm}
	\begin{tabular}{ccc}
		& Index & Term \\
		1st term: & $1$ & $a_1 = 2$\\
		2nd term: & $2$ & $a_2 = 4$\\
		3rd term: &  $3$ & $a_3 = 8$\\
		$\vdots$ & $\vdots$ & $\vdots$ \\
		nth term: & $n$ & $a_n = 2^n$\\
	\end{tabular}
\end{center}

\vspace{0.2cm}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{II. How to Represent a Sequence}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Sequences can be represented by either listing out its terms or by using a general term of the sequence. \\

\vspace{0.2cm}

For example, the sequence 
\begin{center}
	$\{a_n\} = 
	\only<2->{\OS{\B{ listed \, terms }}}{
	\only<2->{\OB}{ \{2, 4,  8, 16, 32, ...\} }} = 
	\only<3->{\OS{\B{ General \, term }}}{
	\only<3->{\OB}{ \{2^n\} }}$
\end{center}

\vspace{0.2cm}

Ex: List out the terms of the sequence $\{a_n\} = \left\{\frac{(n-1)^2}{n}\right\}$
\begin{center}
	$\D{\left\{\frac{(n-1)^2}{n}\right\} 
	= 
	\ON<4->{\B{\OS{ n=1 }{ \left\{\frac{(1-1)^2}{1}\right. ,}}} 
	\ON<5->{\B{\OS{ n=2 }{ \frac{(2-1)^2}{2},}}}
	\ON<6->{\B{\OS{ n=3 }{ \frac{(3-1)^2}{3},}}}
	\ON<7->{\B{\OS{ n=4 \qquad }{ \left.\frac{(4-1)^2}{4}, ... \right\}}}}
	\ON<8->{\B{\,\,= \left\{0, \frac{1}{2}, \frac{4}{3}, \frac{9}{4}, ...\right\}}}}$
\end{center}

\vspace{0.2cm}

Ex: Write the sequence $\{a_n\} = \left\{\frac{3}{2},\frac{9}{4},\frac{27}{6},\frac{81}{8},...\right\}$ in general form.

\begin{center}
	$\D{
	\only<9->{\OS{\B{ n=1 }}}{ \left\{\frac{3}{2}\right. }, 
	\only<9->{\OS{\B{ n=2 }}}{ \frac{9}{4} },
	\only<9->{\OS{\B{ n=3 }}}{ \frac{27}{6} },
	\only<9->{\OS{\B{ n=4 \qquad }}}{ \left.\frac{81}{8}, ... \right\}}
	= \only<10->{\OS{\B{ n=1 }}}{ \left\{\frac{
	\ON<11->{\B{3^1}}}{
	\ON<19->{\B{2\cdot1}}}\right. }, 
	\only<12->{\OS{\B{ n=2 }}}{ \frac{
	\ON<13->{\B{3^2}}}{
	\ON<20->{\B{2\cdot2}}} },
	\only<14->{\OS{\B{ n=3 }}}{ \frac{
	\ON<15->{\B{3^3}}}{
	\ON<21->{\B{2\cdot3}}} },
	\only<16->{\OS{\B{ n=4 \qquad }}}{ \left.\frac{
	\ON<17->{\B{3^4}}}{
	\ON<22->{\B{2\cdot4}}}, ... \right\}}
	= \left\{\frac{
	\ON<18->{\B{3^n}}}{
	\ON<23->{\B{2\cdot n}}}\right\}
	}$
\end{center}

\vspace{0.5cm}
\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}

\begin{frame}
\frametitle{IV. Convergence of Sequences and Properties of Limits}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

A sequence $\{a_n\}$ converges if $\D{\lim \limits_{n \to \infty}\, a_n}$ exists and is finite.  $\{a_n\}$ diverges otherwise.

\vspace{0.2cm}

\EX $\D{\left\{\frac{n}{n+1}\right\}}$ \quad converges \quad since \quad 
$\D{	\lim \limits_{n \to \infty}\, \frac{n}{n+1} 
\ON<2->{\B{\, = \lim \limits_{n \to \infty}\, \frac{\frac{n}{n}}{\frac{n}{n}+\frac{1}{n}}}} 
\ON<3->{\B{\, = \lim \limits_{n \to \infty}\, \frac{1}{1+\frac{1}{n}} = 1}}}$

\vspace{0.4cm}

\EX $\D{\left\{5 - \frac{1}{n^2}\right\}}$ \quad
\ON<5->{\B{converges}} \quad since \quad
$\D{
\lim \limits_{n \to \infty}\, 5 - \frac{1}{n^2}
\ON<4->{\B{\,\, = 5}}}$

\vspace{0.4cm}

\EX $\D{\left\{\ln\left(\frac{1}{n}\right)\right\}}$ \quad
\ON<9->{\B{diverges}} \quad since \quad
$\D{
\lim \limits_{n \to \infty}\, \ln\left(\frac{1}{n}\right) 
\ON<6->{\B{\,\, = \ln\left( \lim \limits_{n \to \infty}\, \frac{1}{n}\right)}}
\ON<7->{\B{\,\,= \ln(0^+)}}
\ON<8->{\B{\,\,= -\infty}}}$

\vspace{0.4cm}

\EX $\D{\left\{\sin\left(\frac{1}{n}\right)\right\}}$ \quad
\ON<13->{\B{converges}} \quad since \quad
$\D{
\lim \limits_{n \to \infty}\, \sin\left(\frac{1}{n}\right) 
\ON<10->{\B{\,\, = \sin\left( \lim \limits_{n \to \infty}\, \frac{1}{n}\right)}}
\ON<11->{\B{\,\,= \sin(0^+)}}
\ON<12->{\B{\,\,= 0}}}$

\vspace{0.6cm}

Let $\lim \limits_{n \to \infty} a_n = A$, $\,\,\lim \limits_{n \to \infty} b_n = B$, and c some constant.

\vspace{0.2cm}

$\D{
(1) \lim \limits_{n \to \infty} (a_n \pm b_n) = 
\ON<14->{\B{\lim \limits_{n \to \infty} a_n \pm \lim \limits_{n \to \infty} b_n}}}$

\vspace{0.2cm}

$\D{
(2) \lim \limits_{n \to \infty} (c \cdot a_n) = 
\ON<15->{\B{c \cdot \lim \limits_{n \to \infty} a_n}}}$

\vspace{0.2cm}

$\D{
(3) \lim \limits_{n \to \infty} (a_n \cdot b_n) = 
\ON<16->{\B{\lim \limits_{n \to \infty} a_n \cdot \lim \limits_{n \to \infty} b_n}}}$

$\D{
(4) \lim \limits_{n \to \infty} \left(\frac{a_n}{b_n}\right) = 
\ON<17->{\B{\frac{\lim \limits_{n \to \infty} a_n}{\lim \limits_{n \to \infty} b_n}}} \hspace{1.0cm} \left(\text{as long as } \,\, 
\ON<18->{\B{\lim \limits_{n \to \infty} b_n \ne 0}}\right)}$

\vspace{0.5cm}

%\textcolor{black}{{\bf Theorem}} :\quad If \quad $\D{\lim \limits_{n \to \infty}\, |a_n| = 0}$ \quad then \quad $\D{\lim \limits_{n \to \infty}\, a_n = 0}$

\end{column}

\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\vspace{0.2cm}

\begin{columns}
\begin{column}{11.0cm}
\EX Determine the convergence of $\D{\left\{\frac{\sin(n)}{n}\right\}}$\\

\vspace{0.4cm}

\ON<2->{\B{This limit cannot be approached directly.  The fact that it contains a $\sin (n)$ term indicates that the squeeze theorem may be useful }}

\vspace{0.2cm}

\ON<3->{\B{For any $n=1,2,3,...$}}
\begin{center}
	$\D{
	\ON<5->{\B{-1}} 
	\ON<4->{\B{\,\, \le \sin n \le \,\,}}
	\ON<6->{\B{1}}}$
	
	\vspace{0.2cm}
	
	$\D{
	\ON<8->{\B{\frac{-1}{n}}} 
	\ON<7->{\B{\,\, \le \frac{\sin n}{n} \le \,\,}} 
	\ON<9->{\B{\frac{1}{n}}}}$
	
	\vspace{0.2cm}
	
	$\D{
	\ON<11->{\B{\lim \limits_{n \to \infty}\, \frac{-1}{n}}}
	\ON<10->{\B{\,\, \le \lim \limits_{n \to \infty} \, \frac{\sin n}{n} \le \,\,}}
	\ON<12->{\B{\lim \limits_{n \to \infty}\, \frac{1}{n}}}}$
	
	\vspace{0.2cm}
	
	$\D{
	\ON<14->{\B{0}}
	\ON<13->{\B{\,\, \le \lim \limits_{n \to \infty} \, \frac{\sin n}{n} \le \,\,}}
	\ON<15->{\B{0}}}$
\end{center}

Therefore

\vspace{0.2cm}

\ON<16->{\B{$\D{\lim \limits_{n \to \infty} \frac{\sin(n)}{n} = 0}$ }}

\vspace{0.2cm}

\quad $\implies$ \quad $\D{\left\{\frac{\sin(n)}{n}\right\}}$ \quad 
\ON<17->{\B{converges}}
