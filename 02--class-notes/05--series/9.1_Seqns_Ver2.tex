\begin{frame}
\frametitle{I. Sequences}
\scriptsize
A \underline{sequence}, $\{a_n\}$, is a \onslide<2->{\textcolor{blue}{finite or infinite list of \underline{terms}}} separated by \onslide<3->{\textcolor{blue}{commas}}. \\

\vspace{0.5cm}

Ex: $\{a_n\} = \{2, 4,  8, 16, 32, ...\}$ \\

\vspace{0.8cm}

Note: In this class, we will only consider terms which are numbers. \\

\vspace{0.8cm}

Each term is associated with an \onslide<4->{\textcolor{blue}{index.}}

\vspace{0.5cm}

Ex: \\

\hspace{1.0cm} $\circ$ The 1st term of \{a\} is 2 \quad (index: $i=1$)\\
\hspace{1.0cm} $\circ$ The 2nd term of \{a\} is 4 \quad (index: $i=2$)\\
\hspace{1.0cm} $\circ$ The 3rd term of \{a\} is 8 \quad (index: $i=3$)\\
\hspace{1.0cm} $\circ$ etc ...

\vspace{0.8cm}

We often refer to the terms of $\{a_n\}$ as \\
\vspace{0.2cm}
$\D{a_1 = 2, \,\,a_2 = 4, \,\,a_3 = 8, \,\,...}$

\vspace{0.4cm}

\end{frame}

\begin{frame}
\frametitle{II. How to Represent a Sequence}
\scriptsize

Sequences can be represented by either listing out its terms or by using a general term of the sequence. \\

\vspace{0.2cm}

For example, the sequence 
\begin{center}
	$\{a_n\} = \{2, 4,  8, 16, 32, ...\}$
\end{center}
can also be represented by
\begin{center}
	$\{a_n\} = \{2^n\}$	
\end{center}

Ex: List out the terms of the sequence $\{a_n\} = \left\{\frac{(n-1)^2}{n}\right\}$

\vspace{1.5cm}

Ex: Write the sequence $\{a_n\} = \left\{\frac{3}{2},\frac{9}{4},\frac{27}{6},\frac{81}{8},...\right\}$ in general form.

\vspace{1.5cm}
\end{frame}

\begin{frame}
\frametitle{III. Plotting and Convergence of a Sequence}
\scriptsize

\vspace{0.2cm}

Consider the sequence with general term \quad $\D{a_n = \frac{n}{n+1}}$

\vspace{0.2cm}

If we think of $a$ as a function of the index $n$, then we can:

%-------LIST---------
\begin{itemize}
	\item[1)] plot the sequence
	\begin{center}
		\includegraphics[scale = 0.2]{\FIGSDIR/9_1_Fig1.png}	
	\end{center}
	\vspace{0.2cm}
	\item[2)] consider the limit of the sequence (as $n \to \infty$)
\begin{center}
$\D{
	\lim \limits_{n \to \infty}\, \frac{n}{n+1} \onslide<2->{\textcolor{blue}{\, = \lim \limits_{n \to \infty}\, \frac{\frac{n}{n}}{\frac{n}{n}+\frac{1}{n}}}} \onslide<3->{\textcolor{blue}{\, = \lim \limits_{n \to \infty}\, \frac{1}{1+\frac{1}{n}} = 1}}
	}$
	
	\vspace{0.2cm}
	
	or
	
	\vspace{0.2cm}
	
	$\D{
	\lim \limits_{n \to \infty}\, \frac{n}{n+1} 
	\ON<4->{\B{\, \US{L'H}{\OS{\left(\frac{\infty}{\infty}\right)}{=}} \lim \limits_{n \to \infty}\, \frac{1}{1} }} 
	\ON<5->{\B{\, = 1}}
	}$	
\end{center}
\end{itemize}
%-------------------

\end{frame}

\begin{frame}
\frametitle{IV. Properties of Sequences}
\scriptsize

Let $\lim \limits_{n \to \infty} a_n = A$, $\,\,\lim \limits_{n \to \infty} b_n = B$, and c some constant.

\begin{align*}
&\\
\quad &(1) \lim \limits_{n \to \infty} (a_n \pm b_n) = \onslide<2->{\textcolor{blue}{\lim \limits_{n \to \infty} a_n \pm \lim \limits_{n \to \infty} b_n}}  \\
&\\
&\\
\quad &(2) \lim \limits_{n \to \infty} (c \cdot a_n) = \onslide<3->{\textcolor{blue}{c \cdot \lim \limits_{n \to \infty} a_n}} \\
&\\
&\\
\quad &(3) \lim \limits_{n \to \infty} (a_n \cdot b_n) = \onslide<4->{\textcolor{blue}{\lim \limits_{n \to \infty} a_n \cdot \lim \limits_{n \to \infty} b_n}} \\
&\\
&\\
\quad &(4) \lim \limits_{n \to \infty} \left(\frac{a_n}{b_n}\right) = \onslide<5->{\textcolor{blue}{\frac{\lim \limits_{n \to \infty} a_n}{\lim \limits_{n \to \infty} b_n}}} \hspace{1.0cm} \left(\text{as long as } \,\, \lim \limits_{n \to \infty} b_n \ne 0\right)
\end{align*}

\vspace{0.5cm}

%\textcolor{black}{{\bf Theorem}} :\quad If \quad $\D{\lim \limits_{n \to \infty}\, |a_n| = 0}$ \quad then \quad $\D{\lim \limits_{n \to \infty}\, a_n = 0}$

\end{frame}
