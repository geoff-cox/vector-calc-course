\begin{frame}
\frametitle{I. Limit Comparison Test}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
	\begin{columns}\fcolorbox{blue}{white}{\begin{column}{9.4cm}
	% Type the contents of the box here
	\underline{Limit Comparison Test (LCT)}:
	\quad Suppose \\
	\vspace{0.2cm}
	%-------LIST---------
	\begin{itemize}
		\item[1.] $a_n > 0,\,\, b_n > 0, \,\,$ (``eventually'')
		\vspace{0.2cm}
		\item[2.] $\D{\lim \limits_{n \to \infty}\, \frac{a_n}{b_n} = L}$ \quad where \quad $0 < L < \infty$
	\end{itemize}
	%-------------------
	%-------LIST---------
	\begin{center}
		then \quad $\D{\sum \limits_{n=1}^{\infty} a_n \quad \text{and} \quad \sum \limits_{n=1}^{\infty} b_n}$ \quad EITHER both converge OR both diverge \\
	\end{center}
	%-------------------
	\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}
	\vspace{0.2cm}
	Determine the convergence or divergence\\
	\vspace{0.2cm}
	$\D{\sum \limits_{n=1}^{\infty} \frac{1}{n^2 - 1}}$
	\quad \ON<2->{\B{let \quad $\D{a_n = \frac{1}{n^2 - 1} > 0}$ \quad and \quad $\D{b_n = \frac{1}{n^2} > 0}$}} \\
	compute \quad $\D{\ON<3->{\B{\lim \limits_{n \to \infty}\, \frac{a_n}{b_n} = \lim \limits_{n \to \infty}\, a_n \cdot \frac{1}{b_n} = }} \ON<4->{\B{\lim \limits_{n \to \infty}\, \frac{1}{n^2 - 1} \cdot n^2 = \lim \limits_{n \to \infty}\, \frac{n^2}{n^2 - 1}} = 1 = L}}$\\
	\ON<5->{\B{Since $0 < L < \infty$ \quad and \quad $\D{\sum \limits_{n=1}^{\infty} \frac{1}{n^2}}$}} \quad \ON<6->{\B{converges by the $p$-test ($p=2>1$) then}} \ON<7->{\B{by the LCT}} \\
	\ON<7->{\B{\fbox{$\D{\sum \limits_{n=1}^{\infty} \frac{1}{n^2 - 1}}$ converges}}}
	
\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{II. Choosing $b_n$}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
	%Suppose we want to determine the convergence or divergence of \qquad $\D{\sum \limits_{n=1}^{\infty} a_n}$ \\
	The key to effectively applying the Limit Comparison Test (LCT) is to \ON<2->{\B{chose the correct $b_n$}} \\
	\vspace{0.2cm}
	The ideal qualities of $b_n$ should be such that
	%-------LIST---------
	\begin{itemize}
		\item \ON<3->{\B{$b_n$ that grows at a similar rate as $a_n$ ($b_n$ keeps the dominating terms of $a_n$)}}
		\item \ON<4->{\B{we know whether $\sum \limits_{n=1}^{\infty} b_n$ converges or diverges.}}
	\end{itemize}
	%-------------------
	For each series, chose the $b_n$ you would use for the LCT \\
	\vspace{0.1cm}
	$\D{\sum \limits_{n=1}^{\infty} \frac{1}{3n^2 + 2}} \,\, \qquad $  \,\, $a_n = \, \D{ \frac{1}{3n^2 + 2}}$ \quad and \quad $b_n = $ \ON<5->{\B{$\D{ \frac{1}{3n^2}}$ \quad ($\sum \limits_{n=1}^{\infty} \frac{1}{3n^2}$ converges by $p$-test)}} \\
	\vspace{0.2cm}
	$\D{\sum \limits_{k=1}^{\infty} \frac{1}{\sqrt{k} - 1}} \,\, \qquad $ \,\, $a_k = \, \D{ \frac{1}{\sqrt{k} - 1}}$ \quad and \quad $b_k = $ \ON<6->{\B{$\D{\frac{1}{\sqrt{k}} = \frac{1}{k^{1/2}}}$\quad ($\sum \limits_{k=1}^{\infty} \frac{1}{k^{1/2}}$ diverges by $p$-test)}} \\
	\vspace{0.2cm}
	$\D{\sum \limits_{n=1}^{\infty} \frac{3^n}{4^n + 5}} \,\, \qquad $ \,\, $a_n = \, \D{ \frac{3^n}{4^n + 5}}$ \quad and \quad $b_n = $ \ON<7->{\B{$\D{\frac{3^n}{4^n} = \left(\frac{3}{4}\right)^n}$\quad ($\sum \limits_{n=1}^{\infty} \left(\frac{3}{4}\right)^n$ converges by GST)}} \\
	\vspace{0.2cm}
	$\D{\sum \limits_{n=1}^{\infty} \frac{3}{\sqrt{n^2 - n + 4}}} \qquad a_n = \, \D{ \frac{3}{\sqrt{n^2 - n + 4}}}$ \,\, and \,\, $b_n = $ \ON<8->{\B{$\D{\frac{3}{\sqrt{n^2}} = \frac{3}{|n|} = \frac{3}{n}}$\quad ($\sum \limits_{n=1}^{\infty} \frac{3}{n}$ div by $p$-test)}} \\
	\vspace{0.2cm}
	$\D{\sum \limits_{n=4}^{\infty} \frac{5n-3}{n^3 - 2n^2 + 5}} \,\, \qquad  \,\, a_n = \, \D{ \frac{5n-3}{n^3 - 2n^2 + 5}}$ \quad and \quad $b_n = $ \ON<9->{\B{$\D{\frac{5n}{n^3} = \frac{5}{n^2}}$\quad ($\sum \limits_{n=1}^{\infty} \frac{5}{n^2}$ conv by $p$-test)}} \\
\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}
