\begin{frame}
\frametitle{I. Series Basics - Definition \& Sigma Notation}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

$\circ$ \,\, A series is the sum of the terms of a sequence. \\

\vspace{0.2cm}

\qquad Finite series: \qquad
$\D{a_2+a_3+...+a_{10} \quad \text{ is the sum of the sequence } \quad \{a_2, a_3, ..., a_{10}\}}$

\vspace{0.2cm}

\qquad Infinite series: \qquad
$\D{a_1+a_2+a_3+... \quad \text{ is the sum of the sequence } \quad \{a_1, a_2, ...\}}$

\vspace{0.4cm}

$\circ$ \,\, Therefore every series has a corresponding \quad \ON<2->{\B{sequence}} \quad associated with it. 

\vspace{0.4cm}

$\circ$ \,\, {\bf Sigma Notation} is short hand for writing series \\

\vspace{0.4cm}

\begin{columns}
\begin{column}{2.0cm}
	Finite series: \\
	\vspace{1.2cm}
\end{column}
\begin{column}{5.0cm}
	$\D{a_2+a_3+...+a_{10} \,\,= \ON<3->{\B{\sum \limits_{k=2}^{10} a_k}}}$ \\
	\vspace{0.2cm}
	$\D{\frac{1}{2} + \frac{2}{3} + \frac{3}{4} + \frac{4}{5} = \ON<4->{\B{\sum \limits_{k=1}^{4} \frac{k}{k+1}}}}$
\end{column}
\end{columns}

\vspace{0.2cm}

\begin{columns}
\begin{column}{2.0cm}
	Infinite series: \\
	\vspace{1.2cm}
\end{column}
\begin{column}{5.0cm}
	$\D{a_1+a_2+a_3+... \,\,= \ON<5->{\B{\sum \limits_{k=1}^{\infty} a_k}}}$ \\
	\vspace{0.2cm}
	$\D{\frac{1}{2} + \frac{2}{3} + \frac{3}{4} + \frac{4}{5} + ... = \ON<6->{\B{\sum \limits_{k=1}^{\infty} \frac{k}{k+1}}}}$
\end{column}
\end{columns}

\end{column}

\begin{frame}
\frametitle{II. Series Basics - Partial Sums, Convergence}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

$\circ$ \,\, A very important idea of series is the fact that a series can be written as a sequence. 

\vspace{0.3cm}

$\circ$ \,\, We call this sequence the \quad\ON<2->{\B{Partial Sums}}\quad of the series.

\vspace{0.1cm}

$\circ$ \,\, The idea is to find a way to write \,\, $\D{\sum \limits_{k = 1}^{\infty} a_k}$ \,\, as an sequence $\D{\{S_1, S_2, S_3, ...\}}$\\

\vspace{0.1cm}

$\circ$ \,\, We do this by defining 

\qquad $\D{S_1 = \sum \limits_{k = 1}^{\ON<3->{\B{1}}} a_k = \ON<4->{\B{a_1}}}$ \hspace{4.4cm}(1st - partial sum) \\
\qquad $\D{S_2 = \sum \limits_{k = 1}^{\ON<5->{\B{2}}} a_k = \ON<6->{\B{a_1 + a_2}}}$ \hspace{3.8cm}(2nd - partial sum)\\
\qquad $\D{S_3 = \sum \limits_{k = 1}^{\ON<7->{\B{3}}} a_k = \ON<8->{\B{a_1 + a_2 + a_3}}}$ \hspace{3.2cm}(3rd - partial sum)\\
\qquad \quad $\D{\vdots}$\\
\qquad $\D{S_N = \sum \limits_{k = 1}^{\ON<9->{\B{N}}} a_k = \ON<10->{\B{a_1 + a_2 + a_3 + ... + a_{N-1} + a_N}}}$ \hspace{1.0cm}(nth - partial sum)\\
\qquad \quad $\D{\vdots}$ \\

\vspace{0.2cm}

\qquad $\implies \qquad \{S_1, S_2, S_3, ... \}$ \qquad is the sequence we are looking for

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Therefore
\begin{center}
	$\D{\sum \limits_{k = 1}^{\infty} a_k = \ON<3->{\B{\lim \limits_{N \to \infty}\,}} \ON<2->{\B{\sum \limits_{k=1}^{N} a_k}} \ON<4->{\B{= \lim \limits_{N \to \infty}\, S_N}}}$
\end{center}

$\circ$ \,\, We say that $\D{\sum \limits_{k = 1}^{\infty} a_k}$ \,\,{\bf converges}\,\, if \,\,$\D{\lim \limits_{N \to \infty}\, S_N}$\,\, converges (is a finite number).  \\
\vspace{0.2cm}
Otherwise, we say that the series {\bf diverges.} \\

\vspace{0.2cm}

$\circ$ \,\, For example, consider the series

\vspace{0.2cm}

$\D{\sum \limits_{k=1}^{\infty} 1} = \ON<5->{\B{1 + 1 + 1 + ...}}$

\vspace{0.2cm}

its partial sums are: \\

\vspace{0.2cm}

\ON<2->{\B{}}
$S_1 = \sum \limits_{k=1}^{1} 1 = \ON<6->{\B{1}},\quad 
 S_2 = \sum \limits_{k=1}^{2} 1 = \ON<7->{\B{1+1 = 2}},\quad 
 S_3 = \sum \limits_{k=1}^{3} 1 =\ON<8->{\B{1+1+1 = 3}},$ \\ 
\hspace{1.2cm}(following this pattern) ..., \quad 
$S_N = \sum \limits_{k=1}^{N} =\ON<9->{\B{N}}, \, ...$

\vspace{0.1cm}

so \quad $\D{\sum \limits_{k=1}^{\infty} 1 = \lim \limits_{N \to \infty}\, \sum \limits_{k=1}^{N} 1 = \lim \limits_{N \to \infty}\, N = \ON<10->{\B{\infty}}}$

\vspace{0.2cm}

And we say $\D{\sum \limits_{k=1}^{\infty} 1}$ \ON<11->{\B{diverges.}}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\frametitle{IV. Series Basics - Properties}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Series have the following properties:

\vspace{0.3cm}

%-------NUM LIST---------
\begin{enumerate}
	\item $\D{\sum \limits_{k=1}^{\infty} C a_k = \ON<2->{\B{C \sum \limits_{k=1}^{\infty} a_k}}}$  \qquad ( C is a constant )
	\vspace{0.1cm}
	\item $\D{\sum \limits_{k=1}^{\infty} ( a_k + b_k ) = \ON<3->{\B{\sum \limits_{k=1}^{\infty} a_k + \sum \limits_{k=1}^{\infty} b_k}}}$
	\vspace{0.1cm}
	\item $\D{\sum \limits_{k=1}^{\infty} ( a_k - b_k ) = \ON<4->{\B{\sum \limits_{k=1}^{\infty} a_k - \sum \limits_{k=1}^{\infty} b_k}}}$
\end{enumerate}
%------------------------

\vspace{0.2cm}

For example,

\vspace{0.2cm}

%-------NUM LIST---------
\begin{enumerate}
	\item $\D{\sum \limits_{k=2}^{\infty} -\frac{10}{k^2} \ON<5->{\B{ = -10 \sum \limits_{k=2}^{\infty} \frac{1}{k^2}}}}$
	\vspace{0.1cm}
	\item $\D{\sum \limits_{k=0}^{\infty} \left(\frac{1}{3^k} - \frac{1}{2^k}\right) \ON<6->{\B{= \sum \limits_{k=0}^{\infty} \frac{1}{3^k} - \sum \limits_{k=0}^{\infty} \frac{1}{2^k}}}}$
\end{enumerate}
%------------------------

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{III. Telescoping Series}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Consider the series: \quad
${\sum \limits_{k = 1}^{\infty} \frac{1}{k+1} - \frac{1}{k}}$ \quad ( telescoping series )


\vspace{0.2cm}

The partial sums are:

${S_1=\sum \limits_{k = 1}^{1} \frac{1}{k+1} - \frac{1}{k} = \ON<2->{\B{\overset{k=1}{\frac{1}{2} - \frac{1}{1}}}}}$ \\
${S_2=\sum \limits_{k = 1}^{2} \frac{1}{k+1} - \frac{1}{k} = \ON<3->{\B{\overset{k=1}{\left[\ON<4->{\B{\rlap{/}}}{\frac{1}{2}} - \frac{1}{1}\right]} + \overset{k=2}{\left[\frac{1}{3} - \ON<4->{\B{\rlap{/}}}{\frac{1}{2}}\right]} = \ON<5->{\B{\frac{1}{3} - \frac{1}{1}}}}}}$ \\

${S_3=\sum \limits_{k = 1}^{3} \frac{1}{k+1} - \frac{1}{k} = \ON<6->{\B{\overset{k=1}{\left[\ON<7->{\B{\rlap{/}}}{\frac{1}{2}} - \frac{1}{1}\right]} + \overset{k=2}{\left[\ON<7->{\B{\rlap{/}}}{\frac{1}{3}} - \ON<7->{\B{\rlap{/}}}{\frac{1}{2}}\right]} + \overset{k=3}{\left[\frac{1}{4} - \ON<7->{\B{\rlap{/}}}{\frac{1}{3}}\right]} = \ON<8->{\B{\frac{1}{4} - \frac{1}{1}}}}}}$ \\

${S_4=\sum \limits_{k = 1}^{4} \frac{1}{k+1} - \frac{1}{k} = \ON<9->{\B{\overset{k=1}{\left[\ON<10->{\B{\rlap{/}}}{\frac{1}{2}} - \frac{1}{1}\right]} + \overset{k=2}{\left[\ON<10->{\B{\rlap{/}}}{\frac{1}{3}} - \ON<10->{\B{\rlap{/}}}{\frac{1}{2}}\right]} + \overset{k=3}{\left[\ON<10->{\B{\rlap{/}}}{\frac{1}{4}} - \ON<10->{\B{\rlap{/}}}{\frac{1}{3}}\right]} + \overset{k=4}{\left[\frac{1}{5} - \ON<10->{\B{\rlap{/}}}{\frac{1}{4}}\right]} = \ON<11->{\B{\frac{1}{5} - \frac{1}{1}}}}}}$ \\
\qquad ${\vdots}$  \qquad (Notice the pattern?)\\
\vspace{0.1cm}
${S_n=\sum \limits_{k = 1}^{N} \frac{1}{k+1} - \frac{1}{k} = \ON<12->{\B{\frac{1}{N+1} - \frac{1}{1}}}}$ \\
\vspace{0.1cm}
Therefore,\\
\vspace{0.1cm}
${\sum \limits_{k = 1}^{\infty} \frac{1}{k+1} - \frac{1}{k} = \lim \limits_{N \to \infty}\, \sum \limits_{k=1}^{N} \frac{1}{k+1} - \frac{1}{k} = \ON<13->{\B{\lim \limits_{N \to \infty}\, \frac{1}{N+1} - \frac{1}{1} = 0 - 1 = -1}}}$ \\
\vspace{0.2cm}
So, we say that ${\sum \limits_{k = 1}^{\infty} \frac{1}{k+1} - \frac{1}{k},}$ \,\, \ON<14->{\B{converges}} \,\, and has sum \ON<15->{\B{\, -1.}} \\

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

%~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\vspace{0.2cm}

Determine if the series converge or diverge and find the sum. \\

$\sum \limits_{k = 1}^{\infty} \sqrt{k+1} - \sqrt{k}$ \qquad [\alert{Note: both the `k+1' \& `k' in the sum are indicators of a telescoping series}]\\

Remember: \quad $\sum \limits_{k = 1}^{\infty} \sqrt{k+1} - \sqrt{k} = \ON<2->{\B{\lim \limits_{N \to \infty}\, \sum \limits_{k = 1}^{N} \sqrt{k+1} - \sqrt{k}}}$ \\

\vspace{0.2cm}
So lets consider
\vspace{0.2cm}
	\setlength{\extrarowheight}{0.2cm}
	\begin{tabular}{ccl}
		$\sum \limits_{k = 1}^{N} \sqrt{k+1} - \sqrt{k}$ & = & $\ON<3->{\B{\overset{k=1}{\overbrace{(\sqrt{1+1} - \sqrt{1})}}}} \ON<4->{\B{\,\,+\,\, \overset{k=2}{\overbrace{(\sqrt{2+1} - \sqrt{2})}}}} \ON<5->{\B{\,\,+\,\, \overset{k=3}{\overbrace{(\sqrt{3+1} - \sqrt{3})}} \,\,+\,\, ...}} $ \\
		& & $\ON<6->{\B{ \hspace{2.8cm} ... \,\,+\,\, \overset{k=N-1}{\overbrace{(\sqrt{N-1+1} - \sqrt{N-1})}} \,\,+\,\, \overset{k=N}{\overbrace{(\sqrt{N+1} - \sqrt{N})}}}}$ \\
		& = & \ON<7->{\B{$(\only<8->{\cancel}{\sqrt{2}} - $\only<9->{\fcolorbox{blue}{white}}{$\sqrt{1}$}$) \,\,+\,\, (\only<8->{\cancel}{\sqrt{3}} - \only<8->{\cancel}{\sqrt{2}}) \,\,+\,\, (\only<8->{\cancel}{\sqrt{4}} - \only<8->{\cancel}{\sqrt{3}}) \,\,+\,\, ...$}} \\
		& & \ON<7->{\B{$\hspace{2.8cm} ... \,\,+\,\, (\only<8->{\cancel}{\sqrt{N}} - \only<8->{\cancel}{\sqrt{N-1}}) \,\,+\,\, ($\only<9->{\fcolorbox{blue}{white}}{$\sqrt{N+1}$}$ - \only<8->{\cancel}{\sqrt{N}})$}} \\
		& = & \ON<10->{\B{$-\sqrt{1} \,\,+\,\, \sqrt{N+1}$}} \\
	\end{tabular}
	
	\vspace{0.2cm}
	
So \\

\vspace{0.4cm}

$\sum \limits_{k = 1}^{\infty} \sqrt{k+1} - \sqrt{k} = \lim \limits_{N \to \infty}\, \sum \limits_{k = 1}^{N} \sqrt{k+1} - \sqrt{k} = \ON<11->{\B{\lim \limits_{N \to \infty}\, (-\sqrt{1} + \sqrt{N+1}) = -1 + \infty = \infty}}$ \\

\vspace{0.2cm}

\ON<12->{\B{Since the limit does not converge to a finite number the series \quad $\sum \limits_{k = 1}^{\infty} \sqrt{k+1} - \sqrt{k}$ \quad {\bf diverges}}}
\vspace{4.0cm}

$\D{}$

\vspace{6.5cm}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}

\begin{frame}
\frametitle{IV. Geometric Series}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Consider the series that has the form 

$$
\sum \limits_{k = 0}^{\infty} r^k
$$

where $r$ is some number. 

\vspace{0.4cm}

Such a series is called a {\bf geometric series}.  When does this series converge? diverge?

\vspace{0.4cm}

Like with every series, we must ask whether the partial sums converge:

\begin{center}
	$\D{
	\sum \limits_{k = 0}^{\infty} r^k = \lim \limits_{N \to \infty}\, \sum \limits_{k = 0}^{N} r^k = \lim \limits_{N \to \infty}\, S_N \qquad \left(\text{What conditions determine the convergence of} \, \{S_N\}? \right)
	}$
\end{center}

To answer this, we must find and analyze $S_N$.  Let's use the following outline:

\vspace{0.3cm}

\hspace{0.8cm} (1) $\D{S_{N+1} = S_N + r^{N+1}}$

\vspace{0.2cm}

\hspace{0.8cm} (2) $\D{S_{N+1} = rS_N + 1}$

\vspace{0.2cm}

\hspace{0.8cm} (3) Find $\D{S_{N}}$

\vspace{0.2cm}

\hspace{0.8cm} (4) Determine the conditions for which $\D{\lim \limits_{N \to \infty}\, S_{N} }$ converges

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

%~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\setlength{\extrarowheight}{0.4cm}
\begin{tabular}{crccccccccccccc}
	(i) & $S_N$ = 
	$\D{	\sum \limits_{k = 0}^{N} r^k}$ = &
	\ON<2->{\B{$\D{1}$}} & 
	\ON<2->{\B{+}} & 
	\ON<3->{\B{$\D{r}$}} & 
	\ON<3->{\B{+}} & 
	\ON<4->{\B{$\D{r^2}$}} & 
	\ON<4->{\B{+}} & 
	\ON<5->{\B{$\dots$}} & 
	\ON<5->{\B{+}} & 
	\ON<6->{\B{$\D{r^{N-1}}$}} & 
	\ON<6->{\B{+}} & 
	\ON<7->{\B{$\D{r^N}$}} & & \\
	(ii) & $S_{N+1}$ = 
	$\D{	\sum \limits_{k = 0}^{N+1} r^k}$ = &
	\ON<8->{\B{$\D{1}$}} & 
	\ON<8->{\B{+}} & 
	\ON<9->{\B{$\D{r}$}} & 
	\ON<9->{\B{+}} & 
	\ON<10->{\B{$\D{r^2}$}} & 
	\ON<10->{\B{+}} & 
	\ON<11->{\B{$\dots$}} & 
	\ON<11->{\B{+}} & 
	\ON<12->{\B{$\D{r^{N-1}}$}} & 
	\ON<12->{\B{+}} & 
	\ON<13->{\B{$\D{r^N}$}} & 
	\ON<13->{\B{+}} & 
	\ON<14->{\B{$\D{r^{N+1}}$}}\\
	(iii) & $rS_N$ \hspace{0.8cm} = &
	\ON<15->{\B{$\D{}$}} & & 
	\ON<15->{\B{$\D{r}$}} & 
	\ON<16->{\B{+}} & 
	\ON<16->{\B{$\D{r^2}$}} & 
	\ON<17->{\B{+}} & 
	\ON<17->{\B{$\dots$}} & 
	\ON<18->{\B{+}} & 
	\ON<18->{\B{$\D{r^{N-1}}$}} & 
	\ON<19->{\B{+}} & 
	\ON<19->{\B{$\D{r^N}$}} & 
	\ON<20->{\B{+}} & 
	\ON<20->{\B{$\D{r^{N+1}}$}}\\
\end{tabular}

\vspace{1.2cm}

(1) $\D{S_{N+1} = S_N + r^{N+1}}$: \qquad 
\ON<21->{\B{This follows by identifying $S_N$ in (ii)}}

\vspace{0.2cm}

(2) $\D{S_{N+1} = rS_N + 1}$: \qquad 
\ON<22->{\B{This follows by identifying $rS_N$ in (ii)}}

\vspace{0.2cm}

(3) Find $\D{S_{N}}$: \qquad 
\ON<23->{\B{Solving for $S_N$ in (1) and (2) yields:}}
%--------EQN----------
\begin{align*}
	\ON<24->{\B{S_N + r^{N+1} \,= rS_N + 1 \quad \iff \quad }}
	\ON<25->{\B{S_N - rS_N}} &\,
	\ON<26->{\B{= 1 - r^{N+1}}}\\
	\ON<27->{\B{(1 - r)S_N}} &\,
	\ON<28->{\B{= 1 - r^{N+1}}}\\
	\ON<29->{\B{S_N}} &\,
	\ON<30->{\B{= \frac{1 - r^{N+1}}{1 - r}}}
\end{align*}
%----------------------

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

%~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

(4) Determine the conditions for which $\D{\lim \limits_{N \to \infty}\, S_{N} }$ converges

\vspace{0.2cm}

$\D{\lim \limits_{N \to \infty}\, S_{N} = 
\ON<2->{\B{\lim \limits_{N \to \infty}\, \frac{1 - r^{N+1}}{1 - r}}}
\ON<3->{\B{\,\, = \lim \limits_{N \to \infty}\, \left(\frac{1}{1 - r} - \frac{r^{N+1}}{1 - r}\right)}} }$

\vspace{0.2cm}

$\D{ \hspace{3.15cm} 
\ON<4->{\B{\,\,=  \frac{1}{1 - r} - \frac{r}{1 - r}\lim \limits_{N \to \infty}\,r^{N} = }}}$
\ON<5->{\B{$\left\{
\begin{array}{lc}
	\ON<7->{\B{\frac{1}{1 - r} - 0}} & 
	\ON<6->{\B{1 < r < 1}}\\
	\ON<9->{\B{\frac{1}{1 - r} - diverges}} & 
	\ON<8->{\B{otherwise}}\\
\end{array}
\right.$}}

Therefore 
\begin{center}
	$\D{\lim \limits_{N \to \infty}\, S_{N}}$ converges only when 
	\ON<10->{\B{$1 < r < 1$ }}
\end{center}
In summary we have:

\vspace{0.2cm}

\begin{columns}\fcolorbox{blue}{white}{\begin{column}{8.4cm}
	\THM {\em \bf 
	\ON<11->{\B{(Geometric Series Test)}}}

\begin{frame}
\frametitle{V. Conditions for Convergence}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
	
	\begin{columns}\fcolorbox{blue}{white}{\begin{column}{8.4cm}
	\begin{center}
		\THM \quad If $\D{\sum \limits_{k=1}^{\infty} a_k}$ converges then 
		\ON<2->{\B{$\lim \limits_{k \to \infty}\, a_k = 0$}}
	\end{center}
	
	\vspace{0.01cm}

	\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}

	\vspace{0.2cm}
	
	$< Proof >$ WTS: 
	\ON<3->{\B{$\lim \limits_{k \to \infty}\, a_k = 0$}}
	
	\vspace{0.2cm}
	
	We Know: \quad  
	\ON<4->{\B{$\D{\sum \limits_{k=1}^{\infty} a_k}$ \quad converges}} \quad 
	\ON<5->{\B{$\implies$ \quad $\D{\lim \limits_{N \to \infty}\, \sum \limits_{k=1}^{N} a_k}$ \quad or \quad $\D{\lim \limits_{N \to \infty}\, S_N}$ \quad converges}}
	
	\vspace{0.4cm}
	
	\ON<6->{\B{Let $\D{\lim \limits_{N \to \infty}\, S_N = L  \quad (some \, finite\, number)}$}}
	
	\vspace{0.4cm}
	
	\ON<7->{\B{( Since we want $\lim \limits_{k \to \infty}\, a_k = 0$, we would like an expression for $a_k$ )}}
	
	\vspace{0.4cm}
	
	$\ON<8->{\B{a_k = }}
	\ON<9->{\B{\, S_k - S_{k-1}}}$ \hspace{1.0cm} 
	\ON<10->{\B{Since \,\,
	$\left\{
	\begin{array}{rl}
		S_k = & 
		\ON<11->{\B{a_1 + a_2 + a_3 + \dots + a_{k-1} + a_k}} \\
		S_{k-1} = & 
		\ON<12->{\B{a_1 + a_2 + a_3 + \dots + a_{k-1}}} \\
	\end{array}
	\right.
	$}}
	
	\vspace{0.6cm}
	
	$
	\ON<13->{\B{\implies \lim \limits_{k \to \infty}\, a_k = }}
	\ON<14->{\B{\,\lim \limits_{k \to \infty}\,S_k - \lim \limits_{k \to \infty}\,S_{k-1}}}
	\ON<15->{\B{\, = L - L = 0}}$
\end{column}

\end{columns}
\end{frame}
