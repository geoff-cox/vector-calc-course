\begin{frame}
\frametitle{I. Introduction}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
Consider the following infinite series:
$$
\onslide<2->{f(\textcolor{blue}{x}) = } \sum \limits_{n=0}^{\infty} \onslide<1->{\textcolor<2->{blue}{x}}^n
$$
There is one major difference between this series and the ones we've been working with.  This series is a function of $x$ and for every value you plug into $f(x)$, you get an \onslide<3->{\textcolor{blue}{infinite series.}}

\begin{center}
	\setlength{\extrarowheight}{0.3cm}
	\begin{tabular}{| c | p{4.4cm} | c |}
		\hline
		$x=0$ & $\displaystyle{f(0) = \onslide<4->{\textcolor{blue}{\sum \limits_{n=1}^{\infty} (0)^n = \sum \limits_{n=1}^{\infty} 0 = 0}}}$ & \onslide<5->{\textcolor{blue}{converges (by the GST)}} \\
		\hline
		$x=1$ & $\displaystyle{f(1) = \onslide<5->{\textcolor{blue}{\sum \limits_{n=1}^{\infty} (1)^n}}}$ & \onslide<6->{\textcolor{blue}{diverges by the GST}}\\
		\hline
		$x=0.75$ & $\displaystyle{f(0.75) = \onslide<7->{\textcolor{blue}{\sum \limits_{n=1}^{\infty} (0.75)^n = \sum \limits_{n=1}^{\infty} 0 = 0}}}$ & \onslide<8->{\textcolor{blue}{converges by the GST}}\\
		\hline
		$x=-\frac{1}{3}$ & $f\left(-\frac{1}{3}\right) = \onslide<9->{\textcolor{blue}{\displaystyle{\sum \limits_{n=1}^{\infty} \left(-\frac{1}{3}\right)^n}}}$ & \onslide<10->{\textcolor{blue}{converges by the GST}}\\
		\hline
		$x=10$ & $\displaystyle{f(10) = \onslide<11->{\textcolor{blue}{\sum \limits_{n=1}^{\infty} (10)^n}}}$ & \onslide<12->{\textcolor{blue}{diverges by the GST}}\\
		\hline
	\end{tabular}	
\end{center}

One obvious question then is: What values of $x$ make $f(x)$ converge?

\begin{frame}
\frametitle{II. Power Series}
\scriptsize
\vspace{0.1cm}
\begin{columns}
\begin{column}{12.0cm}
	\begin{columns}\fcolorbox{blue}{white}{\begin{column}{10.4cm}
	% Type the contents of the box here
	\underline{Power Series}:
	A infinite series of the form
	$$
	f(x) = \sum \limits_{n=0}^{\infty} a_n(x-c)^n = a_0 + a_1(x-c) + a_2(x-c)^2 + ... + a_n (x-c)^n + ...
	$$
	is called a {\bf power series centered at $c$}, where $c$ is constant. \\
	\vspace{0.1cm}
	\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}
	
	\vspace{0.5cm}
	
For each of the following power series, identify $a_n$ and $c$ in the power series definition. \\

\vspace{0.1cm}

\begin{columns}
\begin{column}{3.0cm}
	%--------EQN----------
	\begin{align*}
		g(x) = \sum \limits_{n=0}^{\infty} n^2 (x-3)^n \\
		\\
		\\
	\end{align*}
	%----------------------

	$a_n = \onslide<2->{\textcolor{blue}{n^2}}$ \quad and \quad $c = \onslide<3->{\textcolor{blue}{3}}$

	\vspace{1.2cm}
	
\end{column}
\begin{column}{4.0cm}
	%--------EQN----------
	\begin{align*}
	h(x) &= \sum \limits_{n=0}^{\infty} (-1)^n \frac{(x+1)^n}{n-1} \\
	&\onslide<5->{\textcolor{blue}{\,= \sum \limits_{n=0}^{\infty} \frac{(-1)^n}{n-1}\left(x-(-1)\right)^n}}
	\end{align*}
	%----------------------

	\vspace{0.2cm}
	
	$a_n = \onslide<6->{\textcolor{blue}{\displaystyle{\frac{(-1)^n}{n - 1}}}}$ \quad and \quad $c = \onslide<7->{\textcolor{blue}{-1}}$

	\vspace{1.2cm}
	
\end{column}
\begin{column}{3.5cm}
	%--------EQN----------
	\begin{align*}
		g(x) &= \sum \limits_{n=0}^{\infty} \frac{(2x)^n}{n^2} \\
		&\onslide<8->{\textcolor{blue}{\, = \sum \limits_{n=0}^{\infty} \frac{2^n x^n}{n^2}}} \onslide<9->{\textcolor{blue}{\, = \sum \limits_{n=0}^{\infty} \frac{2^n}{n^2}x^n}}
	\end{align*}
	%----------------------
	
	\vspace{0.2cm}
	
	$a_n = \onslide<10->{\textcolor{blue}{\displaystyle{\frac{2^n}{n^2}}}}$ \quad and \quad $c = \onslide<11->{\textcolor{blue}{0}}$

	\vspace{1.2cm}
	
\end{column}

\frametitle{III. Interval and Radius of Convergence}
\scriptsize
\vspace{0.1cm}
\begin{columns}
\begin{column}{12.0cm}
	\begin{columns}\fcolorbox{blue}{white}{\begin{column}{10.4cm}
	% Type the contents of the box here
	Consider the power series centered at $c$
	$$
	f(x) = \sum \limits_{n=0}^{\infty} a_n(x-c)^n = a_0 + a_1(x-c) + a_2(x-c)^2 + ... + a_n (x-c)^n + ...
	$$
	%\vspace{0.2cm}
	\underline{Interval of Convergence}:
	The interval that contains the values of $x$ for which $f(x)$ is a convergent infinite series is called the {\bf interval of convergence}.  \\
	\vspace{0.2cm}
	\underline{Radius of Convergence}: 
	Half the width of the interval of convergence.\\
	\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}
