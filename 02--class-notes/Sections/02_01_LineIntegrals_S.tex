\begin{columns}\begin{column}{12.0cm}

{I. Definition \& Properties}

\vspace{0.1cm}

\scriptsize

\centering
%--------TABLE---------
\setlength{\extrarowheight}{-0.1cm}
\begin{tabular}{|p{0.4cm}l|}
	\hline
	\hline
	& \\
	\DEF & A {\bf Line Integral} is integration over a curve, $C$ is given by \\
	& $\D{ 
\text{Vector-valued functions:}
\ON<0>{\B{ \quad {\int \limits_{C} {\bf F}({\bf r}) \cdot d{\bf r}}} =\,} 
\ON<0>{\B{\US{\ON<0>{\B{computing}}}{\UB{\int \limits_{a}^{b} {\bf F}({\bf r}(t)) \cdot {\bf r}'(t) \, dt}} \qquad t \in [a,b]}}  }$ \\
	& $\D{ 
\text{Scalar-valued functions:}
\ON<0>{\B{ \quad {\int \limits_{C}^{\,} g({\bf r})\, ds}} =\,} 
\ON<0>{\B{\OS{\ON<0>{\B{form}}}{\OB{\int \limits_{a}^{b} g({\bf r}(t)) \, |{\bf r}'(t)| \, dt}} \qquad t \in [a,b]}}  }$ \\
	& \\
	\UL{Note}: & The expressions above are valid whether $C$ is a planar or space curve. \\
	& \\
	\hline
	\hline
\end{tabular}
%--------------------

\vspace{0.2cm}

%--------TABLE---------
\setlength{\extrarowheight}{-0.2cm}
\begin{tabular}{|p{1.1cm}lc|}
	\hline
	\hline
	& & \\
	\PROP & Let ${\bf F}$ be vector-valued function and $k$ a constant. & 
	\multirow{2}{*}{\includegraphics[scale = 0.15]{Figures/10_1_Fig1.png}}\\
	&&\\
	\multicolumn{2}{|r}{\OBUL $\D{ \int \limits_{C} k\,{\bf F} \cdot d{\bf r} = 
	\ON<0>{\B{k\int \limits_{C} {\bf F} \cdot d{\bf r}} \quad }}$}  
	\OBUL $\D{ \int \limits_{C} \left({\bf F} + {\bf G}\right) \cdot d{\bf r} = 
	\ON<0>{\B{\int \limits_{C} {\bf F} \cdot d{\bf r} + \int \limits_{C} {\bf G} \cdot d{\bf r}}}}$ & \\
	&& \\
	\multicolumn{2}{|r}{\OBUL $\D{ \int \limits_{C} {\bf F} \cdot d{\bf r} = 
	\ON<0>{\B{\int \limits_{C_1} {\bf F} \cdot d{\bf r} + \int \limits_{C_2} {\bf F} \cdot d{\bf r} \quad (C = C_1 \cup C_2) }}}$} & \\

\scriptsize
\begin{columns}\begin{column}{12.0cm}
\EX If ${\bf F} = [y^2,-x^2]$ and $C: y = 4x^2$ \,\, from \, $(0,0)$ \, to \, $(1,4)$, then compute $\D{ \int \limits_{C} {\bf F} \cdot d{\bf r}}$

\SOL 
\ON<0>{\B{We first need to parametrize $C$:}}  \quad  
\ON<0>{\B{Let $x=t \quad \implies$}} \quad 
\ON<0>{\B{$y=4t^2$ \,\, so \,\, }}

\vspace{0.4cm}

\ON<0>{\B{$\RT = [t,4t^2]$}} \quad 
\ON<0>{\B{and since $x = t$ and the path is from $(0,0)$ to  $(1,4)$ \, then,}} \,\, 
\ON<0>{\B{$0 \le t \le 1$}}

\vspace{0.4cm}

\ON<0>{\B{$\D{\RPT = [1,8t]}$ \quad and}} \quad 
\ON<0>{\B{$\D{{\bf F}({\bf r}(t)) = \,
\ON<0>{\B{[(4t^2)^2,-(t)^2] = [16t^4,-t^2]}}}$}} \quad 
\ON<0>{\B{Therefore}}

\vspace{0.4cm}

$\ON<0>{\B{\D{\int \limits_{C} {\bf F}({\bf r}) \cdot d{\bf r} = \int \limits_{a}^{b} {\bf F}({\bf r}(t)) \cdot {\bf r}'(t) \, dt =\,}} 
\ON<0>{\B{\int \limits_{0}^{1}[16t^4,-t^2] \cdot [1,8t] \, dt = \,}}
\ON<0>{\B{\int \limits_{0}^{1} \left( 16t^4 - 8t^3 \right) \, dt}}}$

\vspace{0.4cm}

\hspace{6.7cm} 
\ON<0>{\B{$\D{= \left.\left( \frac{16}{5}t^5 - \frac{8}{4}t^4 \right)\right|_{0}^{1} }$}}

\vspace{0.4cm}

\hspace{6.7cm}
\ON<0>{\B{$\D{= \left( \frac{16}{5}(1)^5 - \frac{8}{4}(1)^4 \right) - \left( 0 \right) = \frac{6}{5}}$}}

\vspace{2.5cm}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}

\begin{frame}
\begin{columns}\begin{column}{12.0cm}
{II. Work}

\vspace{0.1cm}

\scriptsize

Suppose there was a mass moving through a planar force field, ${\bf F}$, along a path, $C$.

\vspace{0.2cm}

\begin{columns}
\begin{column}{7.0cm}
	\includegraphics[scale = 0.36]{Figures/10_1_Fig3.png}
\end{column}
\begin{column}{4.3cm}
	\centering
	Force field: 
	\vspace{0.2cm}
	
	${\bf F}(x,y) = [x,1]$
	
	\vspace{0.2cm}
	
	Parameterization of $C$: 
	
	\vspace{0.2cm}
	
	${\bf r}(t) = [\cos t, \sin t]$, \,
	$\D{0 \le t \le \frac{\pi}{2}}$
	
	\vspace{0.2cm}
	
	What is the work done in moving this mass along $C$?
	
	\vspace{0.6cm}
	
	Previously, we calculated work using the dot product:
	\begin{center}
		$W = {\bf F} \cdot {\bf r}$
	\end{center}
	
	However, since both the force and displacement change at every point along $C$, we are forced to add up ``many" of ``small works'' at different points(i.e. integrate).
	
	\vspace{0.2cm}
	
\end{column}
\end{columns}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}

\begin{center}
	``Small" Work = 
	\ON<0>{\B{(Force)}}
	\ON<0>{\textcolor{Melon}{(``Small" Displacement) =\,}} 
	\ON<0>{\B{(Force)}}
	\ON<0>{\textcolor{red}{(Rate)}}
	\ON<0>{\textcolor{orange}{(``Small" time)}}
	
	\vspace{0.3cm}
	
	$\D{dW = 
	\ON<0>{\B{{\bf F}\, \cdot\,}} 
	\ON<0>{\textcolor{Melon}{d{\bf r} =\, }} 
	\ON<0>{\B{{\bf F}\, \cdot\,}} 
	\ON<0>{\textcolor{red}{{\bf r}'(t)}}\, 
	\ON<0>{\textcolor{orange}{dt}}}$
	
	\vspace{0.4cm}
	
	Total Work = 
	\ON<0>{\B{Sum all the ``Small Work" along $C$ \, (Line Integral)}}
	
	\vspace{0.3cm}
	
	$\D{W = 
	\ON<0>{\B{\int \limits_{C}^{} \, dW =\,}} 
	\ON<0>{\B{\int \limits_{C} {\bf F} \cdot d{\bf r} =\,}} 
	\ON<0>{\B{\int \limits_{a}^{b} {\bf F}({\bf r}(t)) \cdot {\bf r}'(t) dt}}}$
\end{center}

\vspace{4.2cm}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}

\begin{columns}
\begin{column}{6.0cm}
	\EX ${\bf F} = [x,1]$  with ${\bf r}(t) = [\cos t, \sin t]$, \, $\D{0 \le t \le \frac{\pi}{2}}$
	
	\vspace{0.2cm}
	
	What is the work done to move the object along this path when exposed to {\bf F}?
	
	\vspace{0.2cm}
	
	$\FRT = 
	\ON<0>{\B{[\cos t,1]}}$
	
	\vspace{0.2cm}
	
	$\RPT = 
	\ON<0>{\B{[-\sin t, \cos t]}}$
	
	\vspace{0.2cm}
	
	$\FDRT = 
	\ON<0>{\B{-\sin t \cos t + \cos t}}$
	%--------EQN----------
	\begin{align*}
		\ON<0>{\B{\int \limits_{0}^{\pi/2} \FRT \cdot \RPT \, dt}}  
			& \ON<0>{\B{\, = \int \limits_{0}^{\pi/2} -\sin t \cos t + \cos t \, dt}} \\
			& \ON<0>{\B{\, = \int \limits_{0}^{\pi/2} -\sin t \cos t \, dt + \int \limits_{0}^{\pi/2} \cos t \, dt}}  \\
			& \ON<0>{\B{\, = \int \limits_{t=0}^{t=\pi/2} u \, du + \left. \sin t \right|_{0}^{\pi/2}}}  \\
			& \ON<0>{\B{\, = \left.\frac{u^2}{2}\right|_{t=0}^{t=\pi/2} + 1}}  \\
			& \ON<0>{\B{\, = \left.\frac{\cos^2t}{2}\right|_{0}^{\pi/2} + 1 = -\frac{1}{2} + 1 = \frac{1}{2}}}
	\end{align*}
	
	%----------------------
	
\end{column}
\begin{column}{4.8cm}
	\includegraphics[scale = 0.25]{Figures/10_1_Fig3.png}
	 
	 \vspace{0.2cm}

	 \vspace{0.2cm}
	 
	 \ON<0>{\B{$du = -\sin t \, dt$}}

\end{column}

\frametitle{}
\scriptsize
\begin{columns}\begin{column}{12.0cm}
\EX Let $\D{\bf F} = [xy,2y^2]$ represent a force vector field in the x-y plane.  If a particle moves along a quarter circle from $(2,0)$ \, to \, $(0,2)$, then find the work done on that particle.

Want: 
\ON<0>{\B{$\WORK$}}, \quad Need: 
\ON<0>{\B{A parameterization for $C$}}

\vspace{0.2cm}

\ON<0>{\B{Since $C$ is part of a circle, we should think of polar coordinates:}}

\vspace{0.2cm}

$\D{
\ON<0>{\B{x = 2\cos t}}}$\\
$\D{
\ON<0>{\B{y = 2\sin t}} \quad 
\ON<0>{\B{\implies \quad \RT = [2\cos t, 2\sin t], \quad t \in }}
\ON<0>{\B{\, \left[0, \frac{\pi}{2}\right]}} } $\\

\vspace{0.2cm}

$\FRT = 
\ON<0>{\B{[4\sin t\cos t,8\sin^2 t]}}$
	
\vspace{0.2cm}
	
$\RPT = 
\ON<0>{\B{[-2\sin t, 2\cos t]}}$
	
\vspace{0.2cm}
	
$\FDRT = 
\ON<0>{\B{-8\sin^2 t\cos t + 16\sin^2 t\cos t}} 
\ON<0>{\B{\, = 8\sin^2 t\cos t}}$
%--------EQN----------
\begin{align*}
	\ON<0>{\B{\int \limits_{a}^{b} \FRT \cdot \RPT \, dt}}  
		& \ON<0>{\B{\, = \int \limits_{0}^{\pi/2} 8\sin^2 t\cos t \, dt}} \hspace{2.0cm}
		\ON<0>{\B{\, u=\sin t}} 
		\ON<0>{\B{\, \implies du=\cos t dt}} \\
		& \ON<0>{\B{\, = 8\int \limits_{t=0}^{t=\pi/2} u^2 \, du}}\\
		& \ON<0>{\B{\, = 8\left.\frac{u^3}{3}\right|_{t=0}^{t=\pi/2}}}
		  \ON<0>{\B{\, = \left.\frac{8}{3}\sin^3t\right|_{0}^{\pi/2}}}
		  \ON<0>{\B{\, = \frac{8}{3}\left[\sin^3\left(\frac{\pi}{2}\right) - \sin^3\left(0\right)\right]}}
		  \ON<0>{\B{\, = \frac{8}{3}}}  \\
\end{align*}
%----------------------

\vspace{9.6cm}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}\begin{column}{12.0cm}\EX Suppose a wire has density $\delta(x,y,z) = 3\sqrt{5+x}$.  Thinking of the wire as a curve in space, it could be parameterized by $\RT = t\VI + 2t\VJ + \frac{2}{3}t^{3/2}\VK$ \,where\, $0 \le t \le 2$. Find the mass of the wire.

Note 1: \ON<0>{\B{Since the wire is assumed to be a curve, $\delta$ is a linear density $\D{\left(\frac{Mass}{Length}\right)}$}}


Note 2: \ON<0>{\B{Since $\delta$ is not constant, we must integrate where }}

\vspace{0.2cm}

$\hspace{3.0cm} 
\ON<0>{\B{small \, mass}} 
\ON<0>{\B{\, \longrightarrow dM }}
\ON<0>{\B{\, = \, }}
\ON<0>{\B{\,\delta \, ds \longleftarrow }}
\ON<0>{\B{\, \left(\frac{Mass}{Length}\right) (small \, length)}}$

\vspace{0.2cm}

$\D{\implies \quad Mass \, = \, 
\ON<0>{\B{\int \limits_C \delta \, ds = }}
\ON<0>{\B{\, \int \limits_{0}^{2} \delta(\RT) \, |\RPT|\, dt}}}$

\vspace{0.2cm}
	
	$ 
	\ON<0>{\B{\delta(\RT) = 3\sqrt{5+t}}}$
	
	\vspace{0.2cm}
	
	$ 
	\ON<0>{\B{\RPT = [1,2,t^{1/2}]}}$
	$\qquad   
	\ON<0>{\B{\implies \quad |\RPT| = \sqrt{1^2 + 2^2 + (t^{1/2})^2}}}
	\ON<0>{\B{\, = \sqrt{5 + t}}}$
	%--------EQN----------
	\begin{align*}
		\ON<0>{\B{Mass \, }}  
			 \ON<0>{\B{\, = \, \int \limits_{0}^{2} \delta(\RT) \, |\RPT|\, dt}}
			& \ON<0>{\B{\, = \, \int \limits_{0}^{2} \left(3\sqrt{5+t}\right) \, \left(\sqrt{5+t}\right)\, dt}} \\
			& \ON<0>{\B{\, = \, \int \limits_{0}^{2} 3 \left(5+t\right)\, dt}}  \\
			& \ON<0>{\B{\, = \, \left.(15t + \frac{3}{2}t^2) \right|_{0}^{2}}}
			  \ON<0>{\B{\, = \, 30 + \frac{12}{2} = 36}}   \\
	\end{align*}
	
	%----------------------

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}
