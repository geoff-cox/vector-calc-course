\begin{frame}
\frametitle{I. Line Integral}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
\DEF A {\bf Line Integral} is integration over a curve, $C$.  The typical form of a line integral is given by
	
\begin{center}
$\D{ \ON<0>{\B{\int \limits_{C} {\bf F}({\bf r}) \cdot d{\bf r} =\,}} \ON<0>{\B{\int \limits_{a}^{b} {\bf F}({\bf r}(t)) \cdot {\bf r}'(t) \, dt}}}$	
\end{center}

where ${\bf F}$ is a vector-valued function and \, {\bf r} \, is a parametrization of the curve $C$

\vspace{0.2cm}

\EX If ${\bf F} = [y^2,-x^2]$ and $C: y = 4x^2$ \,\, from \, $(0,0)$ \, to \, $(1,4)$, then compute $\D{ \int \limits_{C} {\bf F} \cdot d{\bf r}}$

\SOL 
\ON<0>{\B{We first need to parametrize $C$:}}  \quad  
\ON<0>{\B{Let $x=t \quad \implies$}} \quad 
\ON<0>{\B{$y=4t^2$ \,\, so \,\, }}

\vspace{0.2cm}

\ON<0>{\B{$\RT = [t,4t^2]$}} \quad 
\ON<0>{\B{and since $x = t$ and the path is from $(0,0)$ to  $(1,4)$ \, then,}} \,\, 
\ON<0>{\B{$0 \le t \le 1$}}

\vspace{0.2cm}

\ON<0>{\B{$\D{\RPT = [1,8t]}$ \quad and}} \quad 
\ON<0>{\B{$\D{{\bf F}({\bf r}(t)) = \,\ON<0>{\B{[(4t^2)^2,-(t)^2] = [16t^4,-t^2]}}}$}} \quad 
\ON<0>{\B{Therefore}}

\vspace{0.2cm}

$\ON<0>{\B{\D{\int \limits_{C} {\bf F}({\bf r}) \cdot d{\bf r} = \int \limits_{a}^{b} {\bf F}({\bf r}(t)) \cdot {\bf r}'(t) \, dt =\,}} 
\ON<0>{\B{\int \limits_{0}^{1}[16t^4,-t^2] \cdot [1,8t] \, dt = \,}}
\ON<0>{\B{\int \limits_{0}^{1} \left( 16t^4 - 8t^3 \right) \, dt}}}$

\vspace{0.2cm}

\hspace{6.7cm} 
\ON<0>{\B{$\D{= \left.\left( \frac{16}{5}t^5 - \frac{8}{4}t^4 \right)\right|_{0}^{1} }$}}

\vspace{0.2cm}

\hspace{6.7cm}
\ON<0>{\B{$\D{= \left( \frac{16}{5}(1)^5 - \frac{8}{4}(1)^4 \right) - \left( 0 \right) = \frac{6}{5}}$}}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{III. Additional Concepts}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\begin{columns}
\begin{column}{7.5cm}

\PROP 

\vspace{0.2cm}

\OBUL $\D{ \int \limits_{C} k\,{\bf F} \cdot d{\bf r} = \ON<0>{\B{k\int \limits_{C} {\bf F} \cdot d{\bf r}}}}$ \quad ($k$ constant)

\vspace{0.2cm}

\OBUL $\D{ \int \limits_{C} \left({\bf F} + {\bf G}\right) \cdot d{\bf r} = \ON<0>{\B{\int \limits_{C} {\bf F} \cdot d{\bf r} + \int \limits_{C} {\bf G} \cdot d{\bf r}}}}$

\vspace{0.2cm}

\OBUL $\D{ \int \limits_{C} {\bf F} \cdot d{\bf r} = \ON<0>{\B{\int \limits_{C_1} {\bf F} \cdot d{\bf r} + \int \limits_{C_2} {\bf F} \cdot d{\bf r}}}}$ \quad ($C_1$ \& $C_2$ make up $C$)

\vspace{0.2cm}
	
\end{column}
\begin{column}{3.0cm}

\includegraphics[scale = 0.16]{Figures/10_1_Fig1.png}

\end{column}
\end{columns}

\THM Any two parameterizations of $C$ with \ON<0>{\B{the same orientation}} yield the same value of \ON<0>{\B{$\D{ \int \limits_{C} {\bf F} \cdot d{\bf r}}$}}.

\THM Let $C_1 \, \& \, C_2$ be paths between points $A$ and $B$. Then, in general

\begin{columns}
\begin{column}{6.0cm}

\hspace{3.0cm}
$\D{ \int \limits_{C_1} {\bf F} \cdot d{\bf r}\, \only<1-6>{\quad} \only<7>{=} \only<0>{\,\B{\ne}\,\,} \int \limits_{C_2} {\bf F} \cdot d{\bf r}}$

\end{column}
\begin{column}{2.0cm}

\vspace{0.1cm}
	\includegraphics[scale = 0.13]{Figures/10_1_Fig2.png}
\vspace{0.1cm}
\end{column}
\end{columns}

\UL{Note}: Soon, we will learn that under certain conditions, equality holds in the above statement.
\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}
