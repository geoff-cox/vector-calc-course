{I. Definition}

\vspace{0.2cm}

\scriptsize

\UL{Vector Operation 2}: Dot Product.  Let \,\, $\VA = \AAA$ \,\, and \,\, $\VB = \BBB$.   Then 
%--------EQN----------
\begin{center}
	$\VA \cdot \VB = \ON<0>{\B{|\VA|\,|\VB|\cos \theta}}$
\end{center}
%----------------------
\OBUL An easier way to compute the dot product of 2 known vectors is to use:
%--------EQN----------
\begin{center}
	$\VA \cdot \VB = \AAA \cdot \BBB = \ON<0>{\B{a_1b_1 + a_2b_2 + a_3b_3}}$
\end{center}

\UL{Geometrical Interpretation}:

	$\VA \cdot \VB = $ \ON<0>{(\B{component of $\VA$ along $\VB$})}\ON<0>{(\textcolor{Melon}{length of $\VB$})}

{II. Additional Concepts}

\vspace{0.2cm}

\scriptsize

\NOTE

\vspace{0.1cm}

\OBUL $\VA,\VB \in \RE^3 \quad \implies \quad \ON<0>{\B{\VA \cdot \VB \in \RE \,\, (scalar)}}$ \\

\vspace{0.7cm}

\PROP Let $\VA, \VB, \VC \in \RE^3$ be non-zero.

\vspace{0.2cm}

\OBUL $\VA \cdot \VA = \ON<0>{\B{|\VA|^2}}$ \quad or equivalently \quad $|\VA| = \ON<0>{\B{\sqrt{\VA \cdot \VA}}}$

\vspace{0.2cm}

\OBUL Commutative: $\VA \cdot \VB = \ON<0>{\B{\VB \cdot \VA}}$

\vspace{0.2cm}

\OBUL Distributive: $(\VA + \VB) \cdot \VC = \ON<0>{\B{\VA \cdot \VC + \VB \cdot \VC}}$

\vspace{0.7cm}

\THM Let $\VA$ and $\VB$ be non-zero vectors.  Then
\begin{center}
	\ON<0>{\B{$\VA \cdot \VB = 0$ \quad iff \quad $\VA$ and $\VB$ are \ON<0>{\B{perpendicular (orthogonal)}}}}
\end{center}

\vspace{0.2cm}

\UL{Important Inequalities}:	

\vspace{0.2cm}

\OBUL Triangle Inequality: \quad \ON<0>{\B{$|\VA + \VB| \le |\VA| + |\VB|$}}

\vspace{0.2cm}

\OBUL Cauchy-Schwarz Inequality: \quad \ON<0>{\B{$|\VA \cdot \VB| \le |\VA||\VB|$}}

\vspace{0.7cm}

{III. Application}

\vspace{0.2cm}

\scriptsize
	
\UL{Work}: In the most basic form (1-Dimension), work on an object is given by:
\begin{center}
	Work = \ON<0>{\B{$\US{(on \, an \, object)}{FORCE} \cdot \US{(object \, moved)}{DISTANCE}$}}
\end{center}
However, this assumes:

\vspace{0.2cm}

\OBUL \ON<0>{\B{Constant Force}}

\vspace{0.2cm}

\OBUL \ON<0>{\B{Force is acting in the same direction as the movement of the object.}}

\vspace{0.2cm}

Now let {\bf F} be a force pulling a wagon and {\bf r} the position of the wagon.
\begin{center}
	\includegraphics[scale = 0.25]{Figures/9_2_Fig6.png}
\end{center}
In this case, the force is not acting in the same direction as the movement of the wagon.

\vspace{0.2cm}

To find the work, we must decompose the force to find out how much force is contributing to the movement.

\vspace{0.2cm}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\begin{center}
	\only<1>{\hspace{1.0cm}}
	\only<1>{\includegraphics[scale = 0.25]{Figures/9_2_Fig7.png}}
	\only<0>{\includegraphics[scale = 0.25]{Figures/9_2_Fig8.png}}
	\only<0>{\includegraphics[scale = 0.19]{Figures/9_2_Fig10.png}}
\end{center}

Since $|{\bf F}|\cos\theta$ is the quantity of the force in the direction of movement and $|{\bf r}|$ is the total distance traveled then the work to pull the wagon is 
\begin{center}
	$Work = (Force)(Distance) = \ON<0>{\B{(|{\bf F}|\cos\theta)(|{\bf r}|) =\,}} \ON<0>{\B{|{\bf F}||{\bf r}|\cos\theta =\,}} \ON<0>{\B{{\bf F} \cdot {\bf r}}}$
\end{center}
Therefore, for any constant force, {\bf F}, the general expression for work done is a \ON<0>{\B{dot product}}
\begin{center}
	\ON<0>{\B{$Work = {\bf F} \cdot {\bf r}$}}
\end{center}	

\EX If the wagon is being pulled by a constant force of $[-5,10]\, N$ and the displacement of the wagon is $[-4,0]\, m$, then what is the work being done on the wagon?
\begin{center}
	$Work = \ON<0>{\B{[-5,10] \cdot [-4,0] =\,}} \ON<0>{\B{(-5)(-4) + (10)(0) = 20 \,\text{N-m}}}$
\end{center}
