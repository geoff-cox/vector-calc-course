\begin{frame}
\frametitle{I. Scalar Function}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\DEF Scalar Function

\vspace{0.2cm}

\OBUL A Scalar Function is a function \ON<0>{\B{whose output is a scalar.}}

\vspace{0.2cm}

\EX 

\vspace{0.2cm}

\OBUL $f(x) = x^2$ \quad is a scalar function since \,\, \ON<0>{\B{$f(2) = (2)^2 = 2 \, \longleftarrow \,(scalar)$}}

\vspace{0.2cm}

\hspace{1.0cm} $x \in \ON<0>{\B{\RE}}$ \, and \, $f(x) \in \ON<0>{\B{\RE}}$ \quad \ON<0>{\B{Easy to Visualize}}

\vspace{0.2cm}

\OBUL $f(x,y) = x - x\sqrt{y}$ \quad is a scalar function since \,\, \ON<0>{\B{$f(1,4) = (1) - (1)\sqrt{(4)} = -1\, \longleftarrow \,(scalar)$}}
\begin{center}
	$(x,y) \in \ON<0>{\B{\RE^2}}$ \, and \, $f(x,y) \in \ON<0>{\B{\RE}}$ \quad \ON<0>{\B{Easy to Visualize}} \quad
	\includegraphics[scale = 0.18]{Figures/9_7_Fig1.png}
\end{center}

\OBUL $f(x,y,z) = \frac{\cos(z)}{x} - x$ \quad is a scalar function since \,\, \ON<0>{\B{$f(-1,0,\pi) = \frac{\cos(\pi)}{(-1)} - (-1) = 2\, \longleftarrow \,(scalar)$}}

\vspace{0.2cm}

\hspace{1.0cm} $(x,y,z) \in \ON<0>{\B{\RE^3}}$ \, and \, $f(x,y,z) \in \ON<0>{\B{\RE}}$ \quad \ON<0>{\B{Difficult to Visualize (4D)}}

\begin{frame}
\frametitle{II. Gradient}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\DEF Gradient

\vspace{0.2cm}

Let $D_f$ be the domain of the scalar function $f(x,y)$ and $D_g$ be the domain of the scalar function $g(x,y,z)$.  If $f$ is differentiable on $D_f$ and $g$ is differentiable on $D_g$ then the gradient of $f(x,y)$ and $g(x,y,z)$ are given by

\vspace{0.2cm}

$$\nabla f = \ON<0>{\B{\left[\PFPX,\PFPY\right] =\,}} \ON<0>{\B{\PFPX \VI + \PFPY \VJ}}$$

\vspace{0.2cm}

$$\nabla g = \ON<0>{\B{\left[\PGPX,\PGPY,\PGPZ\right] =\,}} \ON<0>{\B{\PGPX \VI + \PGPY \VJ + \PGPZ \VK}}$$

\vspace{0.8cm}

\EX $f(x,y) = xy + y$. \quad Find $\GF$

\vspace{0.2cm}

\SOL

\vspace{0.2cm}

$\D{\ON<0>{\B{\PFPX =\,}} 
\ON<0>{\B{y}}}$ \quad 
\ON<0>{\B{and}} \quad $\D{
\ON<0>{\B{\PFPY =\,}} 
\ON<0>{\B{x + 1}}}$ \quad 
\ON<0>{\B{$\implies$}} \quad 
\ON<0>{\B{$\D{\GF = [y,x+1]}$}}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\EX $f(x,y,z) = z\ln(xy^2 + y)$. \quad Find $\GF$ at the point $P(0,1,2)$

\vspace{0.2cm}

\SOL

\vspace{0.2cm}

$\D{\ON<0>{\B{\PFPX =\,}} 
\ON<0>{\B{z\frac{1}{xy^2 + y}\cdot \PPX(xy^2 + y) = \,}} 
\ON<0>{\B{z\frac{1}{xy^2 + y}\cdot y^2 =\,}} 
\ON<0>{\B{\frac{zy^2}{xy^2 + y}}}}$

\vspace{0.2cm}

$\D{\ON<0>{\B{\qquad at \,\, P(0,1,2) \quad \implies \quad \PFPX(P) = \,}}
\ON<0>{\B{\left.\frac{zy^2}{xy^2 + y}\right|_{(0,1,2)} = \,}}
\ON<0>{\B{\frac{(2)(1)^2}{(0)(1)^2 + (1)} = 2}}}$

\vspace{0.5cm}

$\D{\ON<0>{\B{\PFPY = \,}}
\ON<0>{\B{z\frac{1}{xy^2 + y}\cdot \PPY(xy^2 + y) = \,}}
\ON<0>{\B{z\frac{1}{xy^2 + y}\cdot (2xy+1) =\,}}
\ON<0>{\B{\frac{z(2xy+1)}{xy^2 + y}}}}$

\vspace{0.2cm}

$\D{\ON<0>{\B{\qquad at \,\, P(0,1,2) \quad \implies \quad \PFPY(P) = \,}}
\ON<0>{\B{\left.\frac{z(2xy+1)}{xy^2 + y}\right|_{(0,1,2)} =\,}}
\ON<0>{\B{\frac{(2)(2(0)(1)+1)}{(0)(1)^2 + (1)} = 2}}}$

\vspace{0.5cm}

$\D{\ON<0>{\B{\PFPZ = \,}}
\ON<0>{\B{\ln(xy^2 + y) }}}$

\vspace{0.2cm}

$\D{\ON<0>{\B{\qquad at \,\, P(0,1,2) \quad \implies \quad \PFPY(P) = \,}}
\ON<0>{\B{\left.\ln(xy^2 + y)\right|_{(0,1,2)} = \,}}
\ON<0>{\B{\ln((0)(1)^2 + (1)) = \ln(1) = 0}}}$

\vspace{0.5cm}

Therefore, \qquad $\D{\ON<0>{\B{\GF(P) = \left[\PFPX(P),\PFPY(P),\PFPZ(P)\right] = \,}}\ON<0>{\B{[2,2,0]}}}$

\frametitle{IV. Additional Concepts}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
	\THM \OBUL $\GF(P)$ \,\, points in the direction of max increase of \, $f$ \, at the point \,$P$.  \\
	\vspace{0.1cm}
	\hspace{0.73cm} \OBUL $\GF$(P) \,\, is perpendicular to the level curve passing through \,$P$
	
\vspace{0.2cm}

\EX Let $\D{f(x,y) = (x+1)\left(\frac{y}{2}+1\right)}$. Find and graph the vector that points in the direction of \NEWLINE maximum increase of $f$ at the point $P = (1,3)$ and $Q = (1,1)$.

\vspace{0.2cm}

\begin{columns}
\begin{column}{4.2cm}
The vectors we want are: 

\vspace{0.1cm}
 
\hspace{1.4cm} \ON<0>{\B{$\GF(P)$ \,\, \& \,\, $\GF(Q)$}}

\vspace{0.2cm}

So if \, $\D{f(x,y) = (x+1)\left(\frac{y}{2}+1\right)}$ \, then

\vspace{0.4cm}

\ON<0>{\B{$\D{\PFPX = (1)\left(\frac{y}{2}+1\right) = \frac{y}{2}+1 \,\, }$}}

\vspace{0.2cm}

%\ON<0>{\B{\hspace{0.8cm} $\D{\implies \,\, \PFPX(P) = \frac{1}{2}+1 = \frac{3}{2}}$}}
