\vspace{0.6cm}

\begin{columns}
\begin{column}{2.5cm}
	\UL{Surface Tangents}:
	
	\vspace{0.2cm}
	
	Tangent along $C_2$:
	
	\vspace{0.2cm}
	
	$\D{{\bf r}_u = \ON<2->{\B{\left[\frac{\partial x}{\partial u},\frac{\partial y}{\partial u},\frac{\partial z}{\partial u}\right]}}}$
	
	\vspace{0.4cm}
	
	Tangent along $C_1$:
	
	\vspace{0.2cm}
	
	$\D{{\bf r}_v = \ON<3->{\B{\left[\frac{\partial x}{\partial v},\frac{\partial y}{\partial v},\frac{\partial z}{\partial v}\right]}}}$
\end{column}

\vspace{0.6cm}

Unit Normal: \quad $\D{{\bf n} = \ON<5->{\B{\frac{{\bf N}}{|{\bf N}|} = \frac{{\bf r}_u \times {\bf r}_v}{|{\bf r}_u \times {\bf r}_v|}}}}$

	\vspace{0.4cm}
	
	$\D{\implies \quad Area(S) = 
	\ON<2->{\B{\iint \limits_{S} \, dS}}}$ \qquad 
	\ON<2->{\B{What is $dS$?}}
\end{center}
\ON<2->{\B{$dS$ represents the area of the small part of the plane approximating the patch of $S$ (see figure)}}

\vspace{0.2cm}

\ON<2->{\B{Note that the small part is a }}
\ON<2->{\B{parallelogram spanned by the vectors: }}
\ON<2->{\B{${\bf r}_u du$ \, and \, ${\bf r}_v dv$}}

\vspace{0.3cm}

\ON<2->{\B{Therefore, \qquad }}
$\D{
\ON<2->{\B{dS = }}
\ON<2->{\B{\,\,|{\bf r}_u du \times {\bf r}_v dv|}} 
\ON<2->{\B{\,= |{\bf r}_u \times {\bf r}_v|\,du\,dv}}}$ \qquad 
\ON<2->{\B{and so}}

\begin{center}
	$\D{ \quad Area(S) =
	\ON<2->{\B{\,\iint \limits_{S} \, dS }}
	\ON<2->{\B{\,= \iint \limits_{R} \, |{\bf r}_u \times {\bf r}_v|\,du\,dv}}}$
\end{center}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{III. Surface Integrals}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

Let ${\bf F}$ be a vector field.

\vspace{0.7cm}

Line Integral \,\, - \,\, $C$ is a curve

\begin{center}
	$\D{\int \limits_{C} {\bf F}({\bf r}) \cdot d{\bf r} = 
	\ON<2->{\B{\int \limits_{a}^{b} {\bf F}(\RT) \cdot \RPT \, dt}}}$
\end{center}

\vspace{0.2cm}

Surface Integral \,\, - \,\, $S$ is a surface

\begin{center}
	$\D{\int \limits_{S} {\bf F}({\bf r}) \cdot d{\bf A} = 
	\ON<3->{\B{\iint \limits_{S} {\bf F} \cdot \only<4->{\OS{\B{ d{\bf A} }}}{\only<4->{\OB}{ {\bf n} \, dA }} =\,}} 
	\ON<5->{\B{\iint \limits_{S} {\bf F} \cdot \only<6->{\OS{\B{ {\bf n} }}}{\only<6->{\OB}{ \frac{{\bf N}}{\only<7->{\CAN}{|{\bf N}|}}\, }} \only<6->{\OS{\B{ dA }}}{\only<6->{\OB}{ \only<7->{\CAN}{|{\bf r}_u \times {\bf r}_v|} \, du\,dv }} \,=\,}} 
	\ON<8->{\B{\iint \limits_{R} {\bf F}({\bf r}(u,v)) \cdot {\bf N}(u,v) \, du \, dv}}}$
\end{center}

\vspace{0.3cm}

where \, ${\bf N} = {\bf r}_u \times {\bf r}_v$ \,  is the normal vector to the surface and \, ${\bf n} = \frac{{\bf N}}{|{\bf N}|}$ \, is the unit normal. $R$ is the region in the $u,v$-plane (parameter plane).

\vspace{1.2cm}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

Flux is a very important concept when modeling physical phenomena.

\vspace{0.2cm}

\begin{center}
