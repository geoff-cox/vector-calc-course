\begin{frame}
\frametitle{I. Series Basics - Definition \& Sigma Notation}
\scriptsize

$\circ$ \,\, A series is the sum of the terms of a sequence. \\

\vspace{0.2cm}

\qquad Finite series: \qquad
$\displaystyle{a_2+a_3+...+a_{10} \quad \text{ is the sum of the sequence } \quad \{a_2, a_3, ..., a_{10}\}}$

\vspace{0.2cm}

\qquad Infinite series: \qquad
$\displaystyle{a_1+a_2+a_3+... \quad \text{ is the sum of the sequence } \quad \{a_1, a_2, ...\}}$

\vspace{0.4cm}

$\circ$ \,\, Therefore every series has a corresponding \quad \onslide<2->{\textcolor{blue}{sequence}} \quad associated with it. 

\vspace{0.4cm}

$\circ$ \,\, {\bf Sigma Notation} is short hand for writing series \\

\vspace{0.4cm}

\begin{columns}
\begin{column}{2.0cm}
	Finite series: \\
	\vspace{1.2cm}
\end{column}
\begin{column}{5.0cm}
	$\displaystyle{a_2+a_3+...+a_{10} \,\,= \onslide<3->{\textcolor{blue}{\sum \limits_{k=2}^{10} a_k}}}$ \\
	\vspace{0.2cm}
	$\displaystyle{\frac{1}{2} + \frac{2}{3} + \frac{3}{4} + \frac{4}{5} = \onslide<4->{\textcolor{blue}{\sum \limits_{k=1}^{4} \frac{k}{k+1}}}}$
\end{column}
\end{columns}

\vspace{0.2cm}

\begin{columns}
\begin{column}{2.0cm}
	Infinite series: \\
	\vspace{1.2cm}
\end{column}
\begin{column}{5.0cm}
	$\displaystyle{a_1+a_2+a_3+... \,\,= \onslide<5->{\textcolor{blue}{\sum \limits_{k=1}^{\infty} a_k}}}$ \\
	\vspace{0.2cm}
	$\displaystyle{\frac{1}{2} + \frac{2}{3} + \frac{3}{4} + \frac{4}{5} + ... = \onslide<6->{\textcolor{blue}{\sum \limits_{k=1}^{\infty} \frac{k}{k+1}}}}$
\end{column}
\end{columns}

\end{frame}

\begin{frame}
\frametitle{II. Series Basics - Partial Sums}
\scriptsize

$\circ$ \,\, A very important idea of series is the fact that a series can be written as a sequence. 

\vspace{0.3cm}

$\circ$ \,\, We call this sequence the \quad\onslide<2->{\textcolor{blue}{Partial Sums}}\quad of the series.

\vspace{0.1cm}

$\circ$ \,\, The idea is to find a way to write \,\, $\displaystyle{\sum \limits_{k = 1}^{\infty} a_k}$ \,\, as an sequence $\displaystyle{\{S_1, S_2, S_3, ...\}}$\\

\vspace{0.1cm}

$\circ$ \,\, We do this by defining 

\qquad $\displaystyle{S_1 = \sum \limits_{k = 1}^{\onslide<3->{\textcolor{blue}{1}}} a_k = \onslide<4->{\textcolor{blue}{a_1}}}$ \hspace{4.4cm}(1st - partial sum) \\
\qquad $\displaystyle{S_2 = \sum \limits_{k = 1}^{\onslide<5->{\textcolor{blue}{2}}} a_k = \onslide<6->{\textcolor{blue}{a_1 + a_2}}}$ \hspace{3.8cm}(2nd - partial sum)\\
\qquad $\displaystyle{S_3 = \sum \limits_{k = 1}^{\onslide<7->{\textcolor{blue}{3}}} a_k = \onslide<8->{\textcolor{blue}{a_1 + a_2 + a_3}}}$ \hspace{3.2cm}(3rd - partial sum)\\
\qquad \quad $\displaystyle{\vdots}$\\
\qquad $\displaystyle{S_n = \sum \limits_{k = 1}^{\onslide<9->{\textcolor{blue}{n}}} a_k = \onslide<10->{\textcolor{blue}{a_1 + a_2 + a_3 + ... + a_{n-1} + a_n}}}$ \hspace{1.0cm}(nth - partial sum)\\
\qquad \quad $\displaystyle{\vdots}$ \\

\vspace{0.2cm}

\qquad $\implies \qquad \{S_1, S_2, S_3, ... \}$ \qquad is the sequence we are looking for

\end{frame}

\begin{frame}
\scriptsize

Therefore
\begin{center}
	$\displaystyle{\sum \limits_{k = 1}^{\infty} a_k = \onslide<3->{\textcolor{blue}{\lim \limits_{n \to \infty}\,}} \onslide<2->{\textcolor{blue}{\sum \limits_{k=1}^{n} a_k}} \onslide<4->{\textcolor{blue}{= \lim \limits_{n \to \infty}\, S_n}}}$
\end{center}

$\circ$ \,\, We say that $\displaystyle{\sum \limits_{k = 1}^{\infty} a_k}$ \,\,{\bf converges}\,\, if \,\,$\displaystyle{\lim \limits_{n \to \infty}\, S_n}$\,\, converges (is a finite number).  \\
\vspace{0.2cm}
Otherwise, we say that the series {\bf diverges.} \\

\vspace{0.2cm}

$\circ$ \,\, For example, consider the series

\vspace{0.2cm}

$\displaystyle{\sum \limits_{k=1}^{\infty} 1} = \onslide<5->{\textcolor{blue}{1 + 1 + 1 + ...}}$

\vspace{0.2cm}

its partial sums are: \\

\vspace{0.2cm}

$S_1 = \sum \limits_{k=1}^{1} 1 = \onslide<6->{\textcolor{blue}{1}},\quad 
 S_2 = \sum \limits_{k=1}^{2} 1 = \onslide<7->{\textcolor{blue}{1+1 = 2}},\quad 
 S_3 = \sum \limits_{k=1}^{3} 1 =\onslide<8->{\textcolor{blue}{1+1+1 = 3}},$ \\ 
\hspace{1.2cm}(following this pattern) ..., \quad 
$S_n = \sum \limits_{k=1}^{n} =\onslide<9->{\textcolor{blue}{n}}, \, ...$

\vspace{0.1cm}

so \quad $\displaystyle{\sum \limits_{k=1}^{\infty} 1 = \lim \limits_{n \to \infty}\, \sum \limits_{k=1}^{n} 1 = \lim \limits_{n \to \infty}\, n = \onslide<10->{\textcolor{blue}{\infty}}}$

\vspace{0.2cm}

And we say $\displaystyle{\sum \limits_{k=1}^{\infty} 1}$ \onslide<11->{\textcolor{blue}{diverges.}}

\end{frame}

\begin{frame}
\frametitle{IV. Series Basics - Properties}
\scriptsize

Series have the following properties:

\vspace{0.3cm}

%-------NUM LIST---------
\begin{enumerate}
	\item $\displaystyle{\sum \limits_{k=1}^{\infty} C a_k = \onslide<2->{\textcolor{blue}{C \sum \limits_{k=1}^{\infty} a_k}}}$  \qquad ( C is a constant )
	\vspace{0.1cm}
	\item $\displaystyle{\sum \limits_{k=1}^{\infty} ( a_k + b_k ) = \onslide<3->{\textcolor{blue}{\sum \limits_{k=1}^{\infty} a_k + \sum \limits_{k=1}^{\infty} b_k}}}$
	\vspace{0.1cm}
	\item $\displaystyle{\sum \limits_{k=1}^{\infty} ( a_k - b_k ) = \onslide<4->{\textcolor{blue}{\sum \limits_{k=1}^{\infty} a_k - \sum \limits_{k=1}^{\infty} b_k}}}$
\end{enumerate}
%------------------------

\vspace{0.2cm}

For example,

\vspace{0.2cm}

%-------NUM LIST---------
\begin{enumerate}
	\item $\displaystyle{\sum \limits_{k=2}^{\infty} -\frac{10}{k^2} \onslide<5->{\textcolor{blue}{ = -10 \sum \limits_{k=2}^{\infty} \frac{1}{k^2}}}}$
	\vspace{0.1cm}
	\item $\displaystyle{\sum \limits_{n=0}^{\infty} \left(\frac{1}{3^n} - \frac{1}{2^n}\right) \onslide<6->{\textcolor{blue}{= \sum \limits_{n=0}^{\infty} \frac{1}{3^n} - \sum \limits_{n=0}^{\infty} \frac{1}{2^n}}}}$
\end{enumerate}
%------------------------

\end{frame}

\begin{frame}
\frametitle{III. Telescoping Series}
\scriptsize

Consider the series: \quad
${\sum \limits_{k = 1}^{\infty} \frac{1}{k+1} - \frac{1}{k}}$ \quad ( telescoping series )


\vspace{0.2cm}

The partial sums are:

${S_1=\sum \limits_{k = 1}^{1} \frac{1}{k+1} - \frac{1}{k} = \onslide<2->{\textcolor{blue}{\overset{k=1}{\frac{1}{2} - \frac{1}{1}}}}}$ \\
${S_2=\sum \limits_{k = 1}^{2} \frac{1}{k+1} - \frac{1}{k} = \onslide<3->{\textcolor{blue}{\overset{k=1}{\left[\onslide<4->{\textcolor{blue}{\rlap{/}}}{\frac{1}{2}} - \frac{1}{1}\right]} + \overset{k=2}{\left[\frac{1}{3} - \onslide<4->{\textcolor{blue}{\rlap{/}}}{\frac{1}{2}}\right]} = \onslide<5->{\textcolor{blue}{\frac{1}{3} - \frac{1}{1}}}}}}$ \\

${S_3=\sum \limits_{k = 1}^{3} \frac{1}{k+1} - \frac{1}{k} = \onslide<6->{\textcolor{blue}{\overset{k=1}{\left[\onslide<7->{\textcolor{blue}{\rlap{/}}}{\frac{1}{2}} - \frac{1}{1}\right]} + \overset{k=2}{\left[\onslide<7->{\textcolor{blue}{\rlap{/}}}{\frac{1}{3}} - \onslide<7->{\textcolor{blue}{\rlap{/}}}{\frac{1}{2}}\right]} + \overset{k=3}{\left[\frac{1}{4} - \onslide<7->{\textcolor{blue}{\rlap{/}}}{\frac{1}{3}}\right]} = \onslide<8->{\textcolor{blue}{\frac{1}{4} - \frac{1}{1}}}}}}$ \\

${S_4=\sum \limits_{k = 1}^{4} \frac{1}{k+1} - \frac{1}{k} = \onslide<9->{\textcolor{blue}{\overset{k=1}{\left[\onslide<10->{\textcolor{blue}{\rlap{/}}}{\frac{1}{2}} - \frac{1}{1}\right]} + \overset{k=2}{\left[\onslide<10->{\textcolor{blue}{\rlap{/}}}{\frac{1}{3}} - \onslide<10->{\textcolor{blue}{\rlap{/}}}{\frac{1}{2}}\right]} + \overset{k=3}{\left[\onslide<10->{\textcolor{blue}{\rlap{/}}}{\frac{1}{4}} - \onslide<10->{\textcolor{blue}{\rlap{/}}}{\frac{1}{3}}\right]} + \overset{k=4}{\left[\frac{1}{5} - \onslide<10->{\textcolor{blue}{\rlap{/}}}{\frac{1}{4}}\right]} = \onslide<11->{\textcolor{blue}{\frac{1}{5} - \frac{1}{1}}}}}}$ \\
\qquad ${\vdots}$  \qquad (Notice the pattern?)\\
\vspace{0.1cm}
${S_n=\sum \limits_{k = 1}^{n} \frac{1}{k+1} - \frac{1}{k} = \onslide<12->{\textcolor{blue}{\frac{1}{n+1} - \frac{1}{1}}}}$ \\
\vspace{0.1cm}
Therefore,\\
\vspace{0.1cm}
${\sum \limits_{k = 1}^{\infty} \frac{1}{k+1} - \frac{1}{k} = \lim \limits_{n \to \infty}\, \sum \limits_{k=1}^{n} \frac{1}{k+1} - \frac{1}{k} = \onslide<13->{\textcolor{blue}{\lim \limits_{n \to \infty}\, \frac{1}{n+1} - \frac{1}{1} = 0 - 1 = -1}}}$ \\
\vspace{0.2cm}
So, we say that ${\sum \limits_{k = 1}^{\infty} \frac{1}{k+1} - \frac{1}{k},}$ \,\, \onslide<14->{\textcolor{blue}{converges}} \,\, and has sum \onslide<15->{\textcolor{blue}{\, -1.}} \\

\end{frame}

\begin{frame}
\frametitle{IV. Geometric Series}
\scriptsize

Consider the series that has the form 
$$
\sum \limits_{k = 0}^{\infty} r^k
$$
where $r$ is some number. 

\vspace{0.2cm}

Such a series is called a {\bf geometric series}.  We determine the convergence or divergence of a geometric series using the following test

\begin{center}
	if \quad $|r| < 1$ \quad then \quad $\sum \limits_{k = 0}^{\infty} r^k$ \quad \onslide<2->{\textcolor{blue}{converges}}\\
	\vspace{0.2cm}
	if \quad $|r| \ge 1$ \quad then \quad $\sum \limits_{k = 0}^{\infty} r^k$ \quad \onslide<3->{\textcolor{blue}{diverges\,\,\,\,}}
\end{center}	

This test is called the {\bf Geometric Series Test}.

\vspace{0.2cm}

Assuming a geometric series converges, we can also find the sum:

\begin{center}
	$\displaystyle{\sum \limits_{k = 0}^{\infty} r^k = \frac{1}{1-r}}$ \,\, ( \alert{This formula is true ONLY if k starts at 0!} )	
\end{center}

\end{frame}

{\bf The Divergence Test}

\begin{center}
	If \quad \onslide<2->{\textcolor{blue}{$\lim \limits_{k \to \infty} a_k \ne 0$}}\quad then \quad \onslide<3->{\textcolor{blue}{$\displaystyle{\sum \limits_{k = 1}^{\infty} a_k}$ \quad diverges}}	
\end{center}
\end{column}}\begin{column}{0.0cm}\end{column}\end{columns}

\vspace{0.4cm}

$\circ$ \,\, For example,
\begin{center}
	$\sum \limits_{n=2}^{\infty} \frac{n+1}{2n-1}$ \quad diverges by the divergence test since \quad $\lim \limits_{n \to \infty}\,\frac{n+1}{2n-1} \overset{\left(\frac{\infty}{\infty}\right)}{\overset{LH}{=}} \lim \limits_{n \to \infty}\,\frac{1}{2} = \frac{1}{2} \ne 0$ \\
\vspace{0.4cm}
	but, for \quad $\sum \limits_{n=2}^{\infty} \frac{1}{2n-1}$ \quad the divergence test is \underline{inconclusive} since \quad $\lim \limits_{n \to \infty}\,\frac{1}{2n-1} = 0$ \\
\end{center}
