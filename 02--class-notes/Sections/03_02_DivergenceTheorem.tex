\UL{I. Introduction}

\scriptsize

\vspace{0.2cm}

To motivate this section, lets start with a problem

\vspace{0.2cm}

Compute the flux of ${\bf F} = [x^2,0,z^2]$ \, through the rectangular box: $|x| \le 1$, \, $|y| \le 3$, \, and $0 \le z \le 2$. \, Let $S$ be the surface of the box.

\vspace{0.2cm}

Note: Since $S$ is composed of 6 surfaces, we need to compute the flux through each surface.

%$\circ$ \,\, The Good News: \B{Sides of rectangular boxes do not need to be parameterized to find the flux.}

\vspace{0.2cm}
 
$\D{Flux = \iint \limits_{S} {\bf F} \cdot {\bf n}\, dA}$

$\D{ \qquad = \iint \limits_{S_{x-}} {\bf F} \cdot {\bf n}_{x-}\, dA + \iint \limits_{S_{x+}} {\bf F} \cdot {\bf n}_{x+}\, dA + \iint \limits_{S_{y-}} {\bf F} \cdot {\bf n}_{y-}\, dA + \iint \limits_{S_{y+}} {\bf F} \cdot {\bf n}_{y+}\, dA + \iint \limits_{S_{z-}} {\bf F} \cdot {\bf n}_{z-}\, dA + \iint \limits_{S_{z+}} {\bf F} \cdot {\bf n}_{z+}\, dA}$

$\D{\iint \limits_{S_{y-}} {\bf F} \cdot {\bf n}_{y-}\, dA = 
\ON<8->{\B{\iint \limits_{S_{y-}} [x^2,0,z^2] \cdot [0,-1,0]\, dA \,=\,}}
\ON<9->{\B{\iint \limits_{S_{y-}} 0 \, dA \, = \, 0 }}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{y+}} {\bf F} \cdot {\bf n}_{y+}\, dA = 
\ON<10->{\B{\iint \limits_{S_{y+}} [x^2,0,z^2] \cdot [0,1,0]\, dA \,=\,}}
\ON<11->{\B{\iint \limits_{S_{y+}} 0 \, dA \, = \, 0 }}}$

\ON<14->{\B{\iint \limits_{S_{z-}} 0 \, dA \,=\, 0}}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{z+}} {\bf F} \cdot {\bf n}_{z+}\, dA = 
\ON<15->{\B{\iint \limits_{S_{z+}} [x^2,0,z^2] \cdot [0,0,1]\, dA \,=\, }}
\ON<16->{\B{\iint \limits_{S_{z+}} z^2 \, dA \OS{\B{z=2}}{\quad=\quad} }}
\ON<17->{\B{\iint \limits_{S_{z+}} 4 \, dA = 4Area(S_{z+}) \,=\, 4(12) = 48}}}$

\begin{center}
	$\D{\implies \quad Flux = \ON<18->{\B{-12+12+0+0+0+48 = 48}}}$
\end{center}

\UL{II. Divergence Theorem}

\vspace{0.2cm}

\scriptsize

Recall:  Green's Thm relates a 
\ON<2->{\B{\, planar closed curve \,}} with the 
\ON<3->{\B{\, region it bounds}}

\vspace{0.2cm}

The Divergence Theorem relates a 
\ON<5->{\B{\,\,closed surface\,\,}} with the 
\ON<6->{\B{\,\,solid it bounds}}

\vspace{0.2cm}

\THM Let \,${\bf F} = [M(x,y,z),N(x,y,z),P(x,y,z)]$ \,\,be a vector field that has continuous 1st partial derivatives.  Also, let $T$ be a bounded volume with boundary $S$ (oriented outward) which is imbedded in ${\bf F}$.  Then

\begin{center}
	$\D{\iint \limits_{S} {\bf F} \cdot {\bf n}\, dA =  
\ON<7->{\B{\,\,\, \iiint \limits_{T} \text{div} \,{\bf F} \,dV}}
\ON<8->{\B{\,\, = \iiint \limits_{T} \left(\frac{\PP M}{\PP x} + \frac{\PP N}{\PP y} + \frac{\PP P}{\PP z}\right)\, dV}}
\ON<9->{\B{\,\, = \iiint \limits_{T} \left(M_x + N_y + P_z\right)\, dV}}
}$
\end{center}
\begin{columns}
\begin{column}{5.0cm}
\centering 

	\includegraphics[scale = 0.18]{Figures/10_5_Fig2.png}
	
	\vspace{0.2cm}
	
	Green's Theorem
	
	\vspace{0.2cm}
	
	$\D{\WORK = 
	\ON<4->{\B{\,\GREENCURL}}}$
\end{column}
\begin{column}{5.0cm}
\centering 

	\includegraphics[scale = 0.18]{Figures/10_7_Fig1.png} 

	\vspace{0.2cm}
	
	Divergence Theorem
	
	\vspace{0.2cm}
	
	$\D{\iint \limits_{S} {\bf F} \cdot {\bf n}\, dA =  
\ON<10->{\B{\,\,\, \iiint \limits_{T} \text{div} \,{\bf F} \,dV}}}$

\end{column}

$\D{
\ON<2->{\B{\text{div} \, {\bf F} }}
\ON<3->{\B{\,= (x^2)_x + (0)_y + (z^2)_z}}
\ON<4->{\B{\,= 2x + 2z}}}$
%--------EQN----------
\begin{align*}
	Flux = 
	\ON<5->{\B{\, 
	\only<6->{\OS{\B{ Divergence \, Theorem }}}{
	\only<6->{\OB}{ \iint \limits_{S} {\bf F} \cdot {\bf n}\, dA 
	\, = \iiint \limits_{T} \text{div} {\bf F} \,\, dV}} }} 
	&\ON<7->{\B{\, =\iiint \limits_{T} 2x + 2z \,\, dV}} \\
	&\ON<8->{\B{\, = \int \limits_{0}^{2} \int \limits_{-3}^{3}\int \limits_{-1}^{1} 2x+2z\, dx\,dy\,dz}} \\
	&\ON<9->{\B{\, = 6\int \limits_{0}^{2} \int \limits_{-1}^{1} 2x+2z\, dx\,dz}} \\
	&\ON<10->{\B{\, = 6\int \limits_{0}^{2} \left. (x^2+2xz) \right|_{-1}^{1}\,dz}} \\
	&\ON<11->{\B{\, = 6\int \limits_{0}^{2} (1+2z)-(1-2z) \,dz}} \\
	&\ON<12->{\B{\, = 6\int \limits_{0}^{2} 4z \,dz}}
	 \ON<13->{\B{\, = 6 \left.(2z^2)\right|_{0}^{2}}}
	 \ON<14->{\B{\, = 6 (8-0) = 48}}
\end{align*}

Now we need to choose our coordinates.  Since $S$ is a sphere, we should use spherical coordinates.

\vspace{0.2cm}

$\ON<7->{\B{x = \rho \sin \phi \cos \theta}}
\ON<8->{\B{ \hspace{1.0cm} 0 \le \rho \le 5}} $ \hspace{1.8cm} Remember:
	
\vspace{0.1cm}
	
$\ON<9->{\B{y = \rho \sin \phi \sin \theta}}
\ON<10->{\B{ \hspace{1.0cm} 0 \le \theta \le 2\pi}} \hspace{2.0cm} 
\ON<11->{\B{\cdot \quad x^2+y^2+z^2 = \rho^2}}$	
	
\vspace{0.1cm}
	
$\ON<12->{\B{z = \rho \cos \phi}} \hspace{1.6cm} 
\ON<13->{\B{0 \le \phi \le \pi}} \hspace{2.15cm} 
\ON<14->{\B{\cdot \quad dV = \rho^2 \sin \phi \, d\rho \, d\theta \, d\phi}}$

\begin{align*}
\ON<15->{\B{(*)}}
\ON<16->{\B{\, = \,\iiint \limits_{T} 3(\rho^2) \,\, dV}}
\ON<17->{\B{\, = \,\int \limits_{0}^{\pi} \int \limits_{0}^{2\pi} \, \int \limits_{0}^{5} \, 3(\rho^2) \,\, \rho^2 \sin \phi \, d\rho \, d\theta \, d\phi}}
&\ON<18->{\B{\, = \,3\int \limits_{0}^{\pi} \int \limits_{0}^{2\pi} \, \int \limits_{0}^{5} \, \rho^4 \, \sin \phi \, d\rho \, d\theta \, d\phi}} \\
&\ON<19->{\B{\, = \,3\int \limits_{0}^{\pi} \sin \phi \, d\phi \int \limits_{0}^{2\pi} \, d\theta \int \limits_{0}^{5} \, \rho^4 \, d\rho}} \\

&\ON<21->{\B{\, = \,3 (1 + 1) \left( 2\pi -0\right) \, \left(\frac{1}{5}(5^5-0) \right)}} \\
&\ON<22->{\B{\, = \,12\pi \cdot 5^4}} \\
\end{align*}

\ON<8->{\B{$\D{{\bf n} = [-1,0,0]}$}} \quad and \quad 
\ON<9->{\B{$x = 0$}}

\ON<12->{\B{\, \iint \limits_{C} 0 \, dA }}

\vspace{0.4cm}
