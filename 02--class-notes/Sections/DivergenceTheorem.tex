\begin{frame}
\frametitle{I. Introduction}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
To motivate this section, lets start with a problem

\vspace{0.2cm}

Compute the flux of ${\bf F} = [x^2,0,z^2]$ \, through the rectangular box: $|x| \le 1$, \,\, $|y| \le 3$, \,\, and $0 \le z \le 2$. \quad Let $S$ be the surface of the box.

\vspace{0.1cm}

\vspace{0.2cm}

$\circ$ \,\, The Bad News: \B{Since $S$ is composed of 6 different surfaces, then we need to compute the flux through each surface.}

\vspace{0.2cm}

$\circ$ \,\, The Good News: \B{Sides of rectangular boxes do not need to be parameterized to find the flux.}

\vspace{0.2cm}
 
$\D{Flux = \iint \limits_{S} {\bf F} \cdot {\bf n}\, dA}$

$\D{ \qquad = \iint \limits_{S_{x-}} {\bf F} \cdot {\bf n}_{x-}\, dA + \iint \limits_{S_{x+}} {\bf F} \cdot {\bf n}_{x+}\, dA + \iint \limits_{S_{y-}} {\bf F} \cdot {\bf n}_{y-}\, dA + \iint \limits_{S_{y+}} {\bf F} \cdot {\bf n}_{y+}\, dA + \iint \limits_{S_{z-}} {\bf F} \cdot {\bf n}_{z-}\, dA + \iint \limits_{S_{z+}} {\bf F} \cdot {\bf n}_{z+}\, dA}$

\vspace{0.2cm}

\begin{columns}
\begin{column}{7.0cm}
	$S_{x-}$: \, 
	$x=\ON<3->{\B{-1}}$,\, 
	$\ON<4->{\B{-3}} \le y \ON<4->{\B{\le 3}}$,\, 
	$\ON<5->{\B{0}} \le z \le \ON<5->{\B{2}}$, \, and \, ${\bf n}_{x-} = 
	\ON<6->{\B{[-1,0,0]}}$
	
	\vspace{0.3cm}
	
	$S_{x+}$: \, 
	$x=\ON<7->{\B{1}}$,\, 
	$\ON<8->{\B{-3}} \le y \le \ON<8->{\B{3}}$,\, 
	$\ON<9->{\B{0}} \le z \le \ON<9->{\B{2}}$, \, and \, ${\bf n}_{x+} = 
	\ON<10->{\B{[1,0,0]}}$
	
	\vspace{0.3cm}
	
	$S_{y-}$: \, 
	$y=\ON<11->{\B{-3}}$,\, 
	$\ON<12->{\B{-1}} \le x \le \ON<12->{\B{1}}$,\, 
	$\ON<13->{\B{0}} \le z \le \ON<13->{\B{2}}$, \, and \, ${\bf n}_{y-} = 
	\ON<14->{\B{[0,-1,0]}}$
	
	\vspace{0.3cm}
	
	$S_{y+}$: \, 
	$y=\ON<15->{\B{3}}$,\, 
	$\ON<16->{\B{-1}} \le x \le \ON<16->{\B{1}}$,\, 
	$\ON<17->{\B{0}} \le z \le \ON<17->{\B{2}}$, \, and \, ${\bf n}_{y+} = 
	\ON<18->{\B{[0,1,0]}}$
	
	\vspace{0.3cm}
	
	$S_{z-}$: \, 
	$z=\ON<19->{\B{0}}$,\, 
	$\ON<20->{\B{-1}} \le x \le \ON<18->{\B{1}}$,\, 
	$\ON<21->{\B{-3}} \le y \le \ON<19->{\B{3}}$, \, and \, ${\bf n}_{z-} = 
	\ON<22->{\B{[0,0,-1]}}$

\scriptsize
\begin{columns}
\begin{column}{12.0cm}

$\D{Flux = \iint \limits_{S_{x-}} {\bf F} \cdot {\bf n}_{x-}\, dA + \iint \limits_{S_{x+}} {\bf F} \cdot {\bf n}_{x+}\, dA + \iint \limits_{S_{y-}} {\bf F} \cdot {\bf n}_{y-}\, dA + \iint \limits_{S_{y+}} {\bf F} \cdot {\bf n}_{y+}\, dA + \iint \limits_{S_{z-}} {\bf F} \cdot {\bf n}_{z-}\, dA + \iint \limits_{S_{z+}} {\bf F} \cdot {\bf n}_{z+}\, dA}$

\vspace{0.4cm}

$\D{\iint \limits_{S_{x-}} {\bf F} \cdot {\bf n}_{x-}\, dA = 
\ON<2->{\B{\iint \limits_{S_{x-}} [x^2,0,z^2] \cdot [-1,0,0]\, dA \,=\, }}
\ON<3->{\B{\iint \limits_{S_{x-}} -x^2 \, dA \OS{\B{x=-1}}{\quad=\quad} }}
\ON<4->{\B{\iint \limits_{S_{x-}} -1 \, dA = -Area(S_{x-}) \,=\, -12}}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{x+}} {\bf F} \cdot {\bf n}_{x+}\, dA = 
\ON<5->{\B{\iint \limits_{S_{x+}} [x^2,0,z^2] \cdot [1,0,0]\, dA \,=\, }}
\ON<6->{\B{\iint \limits_{S_{x+}} x^2 \, dA \OS{\B{x=1}}{\quad=\quad} }}
\ON<7->{\B{\iint \limits_{S_{x+}} 1 \, dA = Area(S_{x+}) \,=\, 12}}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{y-}} {\bf F} \cdot {\bf n}_{y-}\, dA = 
\ON<8->{\B{\iint \limits_{S_{y-}} [x^2,0,z^2] \cdot [0,-1,0]\, dA \,=\,}}
\ON<9->{\B{\iint \limits_{S_{y-}} 0 \, dA \, = \, 0 }}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{y+}} {\bf F} \cdot {\bf n}_{y+}\, dA = 
\ON<10->{\B{\iint \limits_{S_{y+}} [x^2,0,z^2] \cdot [0,1,0]\, dA \,=\,}}
\ON<11->{\B{\iint \limits_{S_{y+}} 0 \, dA \, = \, 0 }}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{z-}} {\bf F} \cdot {\bf n}_{z-}\, dA = 
\ON<12->{\B{\iint \limits_{S_{z-}} [x^2,0,z^2] \cdot [0,0,-1]\, dA \,=\, }}
\ON<13->{\B{\iint \limits_{S_{z-}} -z^2 \, dA \OS{\B{z=0}}{\quad=\quad} }}
\ON<14->{\B{\iint \limits_{S_{z-}} 0 \, dA \,=\, 0}}}$

\vspace{0.2cm}

$\D{\iint \limits_{S_{z+}} {\bf F} \cdot {\bf n}_{z+}\, dA = 
\ON<15->{\B{\iint \limits_{S_{z+}} [x^2,0,z^2] \cdot [0,0,1]\, dA \,=\, }}
\ON<16->{\B{\iint \limits_{S_{z+}} z^2 \, dA \OS{\B{z=2}}{\quad=\quad} }}
\ON<17->{\B{\iint \limits_{S_{z+}} 4 \, dA = 4Area(S_{z+}) \,=\, 4(12) = 48}}}$
	
\vspace{0.5cm}

Therefore,
\begin{center}
	$\D{Flux = \ON<18->{\B{-12+12+0+0+0+48 = 48}}}$
\end{center}

Fortunately, there is a better way to do this.

\begin{frame}
\frametitle{II. Diverence Theorem}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

An analogous theorem to Green's Theorem is known as the Divergence Theorem:

\vspace{0.2cm}

\THM Let ${\bf F}(x,y,z)$ be a vector field that has continuous 1st partial derivatives.  Also, let $T$ be a closed bounded volume with boundary $S$ which is imbedded in ${\bf F}$.  Then

\vspace{0.5cm}

\hspace{1.8cm} (General Form) \qquad $\D{\iint \limits_{S} {\bf F} \cdot {\bf n}\, dA = \iiint \limits_{T} 
\ON<2->{\B{\,\,\, \text{div} {\bf F}}} \,\,\, dV}$
	
	\vspace{0.4cm}
	
\hspace{0.2cm} (Component Form) \qquad $\D{\iint \limits_{S} \left(F_1\, dy\,dz + F_2 \, dx\,dz + F_3 \, dx\,dy\right) = \iiint \limits_{T} \left(\frac{\PP F_1}{\PP x} + \frac{\PP F_2}{\PP y} + \frac{\PP F_3}{\PP z}\right)\, dx\,dy\,dz}$

\vspace{0.2cm}

	\includegraphics[scale = 0.2]{Figures/10_7_Fig1.png} 
\end{center}

\end{column}

\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
Back to our first problem:

\vspace{0.2cm}

Compute the flux of ${\bf F} = [x^2,0,z^2]$ \, through the rectangular box: $|x| \le 1$, \,\, $|y| \le 3$, \,\, and $0 \le z \le 2$. \quad Let $S$ be the surface of the box.

\vspace{0.8cm}

$\D{Flux = \only<2->{\OS{\B{ Divergence \, Theorem }}}{\only<2->{\OB}{ \iint \limits_{S} {\bf F} \cdot {\bf n}\, dA = \iiint \limits_{T} \text{div} {\bf F} \,\, dV }} = \,
\ON<3->{\B{\iiint \limits_{T} \text{div} [x^2,0,z^2] \,\, dV = \,}}
\ON<4->{\B{\iiint \limits_{T} 2x + 2z \,\, dV}}}$

$\D{\hspace{1.0cm} 
\ON<5->{\B{= \int \limits_{0}^{2} \int \limits_{-3}^{3}\int \limits_{-1}^{1} 2x+2z\, dx\,dy\,dz}}}$

$\D{\hspace{1.0cm} 
\ON<6->{\B{= \int \limits_{0}^{2} \int \limits_{-3}^{3} \left. (x^2+2xz) \right|_{-1}^{1}\,dy\,dz =\,}}
\ON<7->{\B{\int \limits_{0}^{2} \int \limits_{-3}^{3} (1+2z)-(1-2z) \,dy\,dz = \,}}
\ON<8->{\B{\int \limits_{0}^{2} \int \limits_{-3}^{3} 4z \,dy\,dz}}}$ \quad 
\ON<8->{\B{(integrate $x$)}}

$\D{\hspace{1.0cm} 
\ON<9->{\B{= \int \limits_{0}^{2} \left.(4zy)\right|_{-3}^{3}\,dz =\,}} 
\ON<10->{\B{\int \limits_{0}^{2} (12z)-(-12z)\,dz =\,}} 
\ON<11->{\B{\int \limits_{0}^{2} 24z \,dz}}}$ \hspace{3.1cm} 
\ON<11->{\B{(integrate $y$)}}

$\D{\hspace{1.0cm} 
\ON<12->{\B{= \int \limits_{0}^{2} 12z^2 \,dz = 48-0 = 48}}}$ \hspace{6.0cm} 
\ON<12->{\B{(integrate $z$)}}
	
\vspace{1.0cm}

So, using the divergence theorem, we have \quad $Flux = \ON<13->{\B{48}}$
\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}
