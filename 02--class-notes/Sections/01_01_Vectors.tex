{I. Vector Basics}

\vspace{0.2cm}

\scriptsize

\DEF A {\bf vector}, ${\bf v}$, is a mathematical object that has both \ON<2->{\B{\quad magnitude/length \quad }} and \ON<3->{\B{\quad direction.}}

\vspace{0.5cm}

\UL{Notation}: \\

\vspace{0.2cm}

\OBUL  Vectors will written either in bold or with an arrow: \quad {\bf v}, $\vec{v}$.\\

\vspace{0.2cm}

\OBUL  A vector, $\vec{v}$, in $\mathbf{R}^2$ (\,$\vec{v} \in \mathbf{R}^2$\,) can be represented by \ON<6->{\B{\,2\,}} coordinates: \quad $\vec{v} = \ON<6->{\B{\VV \in \mathbf{R}^2}}$\\

\vspace{0.2cm}

\OBUL  A vector, $\vec{v}$, in $\mathbf{R}^3$ (\,$\vec{v} \in \mathbf{R}^3$\,) can be represented by \ON<7->{\B{\,3\,}} coordinates: \quad $\vec{v} = \ON<7->{\B{\VVV \in \mathbf{R}^3}}$\\

\vspace{0.2cm}

\OBUL  The special unit vectors {\bf i} = [1,0,0], {\bf j} = [0,1,0], and {\bf k} = [0,0,1] form a basis for $\RE^3$, \NEWLINE so we can write \,\, $\vec{v} = \VVV$ \,\, as \,\, \ON<8->{\B{$\vec{v} = v_1{\bf i} + v_2{\bf j} + v_3{\bf k}$ \quad (linear combination of \VI, \VJ, \VK)}}

\vspace{0.5cm}

\EX \\
\vspace{0.1cm}
\OBUL $[10,14], [0,1], [-1,2] \ON<9->{\B{\,\, \in \mathbf{R}^2}}$ \quad and \quad $[10,4] = \ON<10->{\B{10\VI + 4 \VJ}}$\\
\vspace{0.2cm}
\OBUL $[1,1,-1], [0,0,0], [-10,12,-8] \ON<11->{\B{\,\, \in \mathbf{R}^3}}$ \quad and \quad $[-10,12,-8] = \ON<12->{\B{-10\VI + 12 \VJ - 8\VK}}$\\

\vspace{0.5cm}

\THM Let ${\bf a} = [a_1,a_2,a_3]$ and ${\bf b} = [b_1,b_2,b_3]$ be vectors. Then
\begin{center}
	{\bf a} = {\bf b} \,\, $iff$ \quad \ON<13->{\B{$a_1 = b_1$, $a_2 = b_2$, $a_3 = b_3$}}
\end{center}

\EX \quad $[2,0,2d+c] = [c,0,1] \quad \implies$ \quad \ON<14->{\B{$c = 2$ \quad and \quad $2d+c=1$,}} \quad \ON<15->{\B{so \quad $d=-\frac{1}{2}$}}

{II. Vector Addition}

\vspace{0.2cm}

\scriptsize

\UL{Vector Operation 1}: Addition

\vspace{0.2cm}

Let \,\, $\VA = \AAA$ \,\, and \,\, $\VB = \BBB$

\vspace{0.2cm}

Then 
%--------EQN----------
\begin{align*}
	\VA + \VB = \AAA + \BBB = \ON<2->{\B{[a_1+b_1, a_2+b_2, a_3+b_3]}}
\end{align*}
%----------------------

\begin{center}
	\only<1-2>{\includegraphics[scale = 0.19]{Figures/9_1_Fig1.png}}
	\only<3->{\includegraphics[scale = 0.19]{Figures/9_1_Fig2.png}}
	\ON<4->{\B{\includegraphics[scale = 0.19]{Figures/9_1_Fig3.png}}}
\end{center}

\UL{Note}:

\vspace{0.1cm}

\OBUL $\VA,\VB \in \RE^3 \quad \implies \quad \ON<5->{\B{\VA + \VB \in \RE^3}}$ \\

\vspace{0.1cm}

\OBUL $\VA + \VB$ is known as the \ON<6->{\B{\,\,resultant\,\,}} of $\VA$ and $\VB$. (e.g. If $\VA$ and $\VB$ were forces then $\VA + \VB$ would be \NEWLINE the \ON<7->{\B{resultant force.}}

{III. Scalar Multiplication}

\vspace{0.2cm}

\scriptsize

\UL{Property}: Scalar Multiplication of Vectors

\vspace{0.2cm}

Let \,\, $\V = \VVV \in \RE^3$ \,\, and \,\, $c \in \RE$ (scalar)

\vspace{0.2cm}

Then 
%--------EQN----------
\begin{align*}
	c\V = c\VVV = \ON<2->{\B{[cv_1, cv_2, cv_3]}}
\end{align*}
%----------------------

{IV. Length(Norm) of a Vector}

\vspace{0.2cm}

\scriptsize

\DEF Let $\V = [v_1,v_2]$ and $\W = [w_1,w_2,w_3]$.  Then the \UL{length} of $\V$ and $\W$ (denoted $|\V|$ and $|\W|$) is given by
\begin{align*}
	|\V| = \ON<2->{\B{\sqrt{v_1^2 + v_2^2}}} \quad \text{and} \quad |\W| = \ON<3->{\B{\sqrt{w_1^2 + w_2^2 + w_3^2}}}
\end{align*}

\EX Find the length of $\W = [1,-2,4]$
%--------EQN----------
\begin{align*}
	|\W| = \ON<4->{\B{\sqrt{(1)^2 + (-2)^2 + (4)^2} =}} \ON<5->{\B{\sqrt{1+4+16} = \sqrt{21}}}
\end{align*}
%----------------------

\UL{Note}:

\vspace{0.1cm}

\OBUL $\W \in \RE^3$ (vector),\,\, but $|\W| \in $\ON<6->{\B{$\, \RE $ (scalar)}} \\

\vspace{0.4cm}

\DEF $\V$ \,\,\, is a \UL{unit vector} if \,\,\, \ON<7->{\B{$|\V| = 1$}}

\vspace{0.4cm}

\EX Find a unit vector, {\bf u}, in the same direction as \W \, above.
%--------EQN----------
\begin{align*}
	{\bf u} = \ON<8->{\B{\frac{\W}{|\W|} =\,}} \ON<9->{\B{\frac{1}{\sqrt{21}}[1,-2,4] = \left[\frac{1}{\sqrt{21}},\frac{-2}{\sqrt{21}},\frac{4}{\sqrt{21}}\right]}} \\
\end{align*}
