\begin{frame}
\frametitle{Line Integrals where $C$ is closed}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

To compute a line integral \quad $\D{\oint \limits_{C} {\bf F} \cdot d{\bf r}}$ \quad where $C$ is a {\em closed curve} $C$, 

\vspace{0.4cm}

we could use the following guideline:

\vspace{0.4cm}

$\circ$ \quad {\bf Step 1}: \quad 
\ON<2->{\B{Compute curl\,{${\bf F}$}}}

\vspace{0.2cm}

\hspace{0.8cm} - \, if \, 
\ON<3->{\B{curl\,{${\bf F} = 0$}}} \quad then \quad $\oint \limits_{C} {\bf F} \cdot d{\bf r} = 
\ON<4->{\B{0}}$

\vspace{0.2cm}

\hspace{0.8cm} - \, if \, 
\ON<5->{\B{curl\,{${\bf F} \ne 0$}}} \,\, then compute \,\, $\oint \limits_{C} {\bf F} \cdot d{\bf r}$ \,\, either \,
$\left\{
\begin{array}{l}
	\ON<6->{\B{\text{directly}}}\\
	\ON<7->{\B{\text{using Green's Thm ($S$ a plane)}}}\\
	\ON<8->{\B{\text{using Stokes' Thm ($S$ not a plane)}}}
\end{array}
\right.$

\vspace{0.6cm}

$\circ$ \quad {\bf Step 2}: Assuming curl\,{${\bf F} \ne 0$}, decide how to compute \, $\oint \limits_{C} {\bf F} \cdot d{\bf r}$

\begin{columns}
\begin{column}{0.2cm}
\end{column}
\begin{column}{2.2cm}
	i. Directly:
	
	\vspace{1.2cm}
\end{column}
\begin{column}{9.0cm}
	%-------LIST---------
\begin{itemize}
	\item \ON<9->{\B{Parameterize $C$ to find ${\bf r}(t)$ for $t$ on $[a,b]$}}
	\vspace{0.2cm}
	\item \ON<10->{\B{Compute ${\bf F}({\bf r}(t))$ and ${\bf r}'(t)$}}
	\vspace{0.2cm}
	\item \ON<11->{\B{Evaluate $\oint \limits_{C} {\bf F} \cdot d{\bf r}$}}
\end{itemize}
%-------------------
\end{column}
\end{columns}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\begin{columns}
\begin{column}{0.2cm}
\end{column}
\begin{column}{2.2cm}
	ii. Green's Thm:
	
	\vspace{1.6cm}
\end{column}
\begin{column}{9.0cm}
	%-------LIST---------
\begin{itemize}
	\item Find \, 
	\ON<2->{\B{$\frac{\PP F_2}{\PP x}$}} \, and \, 
	\ON<3->{\B{$\frac{\PP F_1}{\PP y}$}}
	\vspace{0.2cm}
	\item Determine 
	\ON<4->{\B{R (in the $x,y$-plane)}}
	\vspace{0.2cm}
	\item Setup and evaluate 
	\ON<5->{\B{$\iint \limits_{R} \left(\frac{\PP F_2}{\PP x} - \frac{\PP F_1}{\PP y}\right)\, dxdy$}}
\end{itemize}
%-------------------
\end{column}
\end{columns}

\vspace{0.8cm}

\begin{columns}
\begin{column}{0.2cm}
\end{column}
\begin{column}{2.1cm}
	iii. Stokes' Thm:
	
	\vspace{3.2cm}
\end{column}
\begin{column}{9.0cm}
	%-------LIST---------
\begin{itemize}
	\item Collect \, 
	\ON<6->{\B{curl\,{${\bf F} = 0$} \, found in Step 1}}
	\vspace{0.2cm}
	\item $\left\{
	\begin{array}{l}
		 \text{If $S$ is a plane,  
		 \ON<7->{\B{don't parameterize $S$ (i.e. don't find ${\bf r}$)}}} \\
		 \text{If $S$ is a not a plane, 
		 \ON<8->{\B{parameterize (i.e. find ${\bf r}$)}}}
	\end{array}\right.$
	\vspace{0.2cm}
	\item $\left\{
	\begin{array}{l}
		 \text{If $S$ is a plane, 
		 \ON<9->{\B{determine ${\bf n}$ (should be clear by observation)}}} \\
		 \text{If $S$ is a not a plane, 
		 \ON<10->{\B{compute ${\bf N} = {\bf r}_u \times {\bf r}_v$}}}
	\end{array}\right.$
	\vspace{0.2cm}
	\item $\left\{
	\begin{array}{l}
		 \text{If $S$ is a plane, 
		 \ON<11->{\B{evaluate $\iint \limits_{S} \text{curl}\,{\bf F} \cdot {\bf n} \, dA$}}} \\
		 \text{If $S$ is a not a plane, 
		 \ON<12->{\B{evaluate $\iint \limits_{S} \text{curl}\,{\bf F} \cdot {\bf N} \, dxdy$}}}
	\end{array}\right.$
\end{itemize}
%-------------------
\end{column}
\end{columns}

\begin{frame}
\frametitle{Surface Integrals where $S$ is not closed}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
When $S$ is not closed (doesn't bound a solid volume), then there is really only one option for computing
	
\vspace{0.4cm}
	
\begin{center}
	$\D{\FLUX}$
\end{center}

\vspace{0.2cm}

\begin{columns}
\begin{column}{6.5cm}
	
\vspace{0.2cm}
	
%-------LIST---------
\begin{itemize}
	\item Find 
	\ON<2->{\B{${\bf r}(u,v)$ and evaluate}}
	\vspace{0.2cm}
	\begin{center}
		\ON<3->{\B{$\D{\iint \limits_{R} {\bf F}({\bf r}) \cdot {\bf N}\, du\,dv}$}}
	\end{center}
	\vspace{0.2cm}
 	\item If $S$ is a 
	\ON<4->{\B{plane, \,\,}} it may be easier
	\ON<5->{\B{not to find ${\bf r}$ and instead evaluate}}
	\vspace{0.2cm}
	\begin{center}
		\ON<6->{\B{$\D{\iint \limits_{R} {\bf F} \cdot {\bf n}\, du\,dv}$}}
	\end{center}
\end{itemize}
%-------------------

\vspace{0.2cm}

\qquad $R$ is the domain of {\bf r} (i.e. u,v-plane region)

%-------LIST---------

	\item \ON<2->{\B{Apply the Divergence Theorem
	\begin{center}
		$\D{\FLUX = \iiint \limits_{T} \text{div} \, {\bf F}\, dV}$
	\end{center}}}
\end{itemize}

%-------LIST---------

	\item[2.] \ON<4->{\B{Setting up and evaluating the triple integral 
	\begin{center}
		$\D{\iiint \limits_{T} \text{div} \, {\bf F}\, dV}$
	\end{center}}}
