\OBUL When representing a line integral over a closed curve, we will use the notation

\begin{center}
	\ON<3->{\B{$\OWORK$}} \quad instead of \quad $\WORK$

\begin{frame}
\frametitle{II. Green's Theorem}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}
\THM Let $R$ be a bounded region in the $xy$-plane with boundary $C$.  If ${\bf F}(x,y) = [F_1, F_2]$ is a vector field, then

\vspace{0.6cm}

\hspace{2.0cm} (General Form) \qquad $\D{\OWORK = \GINT}$
	
	\vspace{0.4cm}
	
\hspace{1.5cm} (Component Form) \qquad $\D{\GREENSL = \GREENSA}$

\vspace{0.6cm}

\begin{center}
	\includegraphics[scale = 0.2]{Figures/10_5_Fig2.png}
\end{center}

\end{column}
\begin{column}{0.0cm}\end{column}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}
\begin{column}{12.0cm}

\EX Let $\D{{\bf F} = [x^2 + 4y, x + y^2]}$ and let \,\, $C$ be a square bounded by: \,\, $x=0$, $x=1$, $y=0$, and $y=1$,

\vspace{0.2cm}

Compute \quad $\D{\OWORK}$

\vspace{0.2cm}

\SOL To compute this directly we would have to compute 
\ON<2->{\B{a line integral for $C_1$, $C_2$, $C_3$, and $C_4$}}

\begin{columns}
\begin{column}{7.5cm}

\vspace{0.3cm}

$\D{\OWORK = 
\ON<3->{\B{\oint \limits_{C_1} \FDR + \oint \limits_{C_2} \FDR + \oint \limits_{C_3} \FDR + \oint \limits_{C_4} \FDR}}}$

\vspace{0.3cm}

\ON<4->{\B{Alternatively, we could apply Green's Theorem}}

\vspace{0.4cm}

\ON<4->{\B{$\D{\OWORK = \GREENSA}$}}

\vspace{0.4cm}

$\D{
\ON<5->{\B{\frac{\partial F_2}{\partial x} = 1}}}$ \quad 
\ON<6->{\B{and}} \quad $\D{
\ON<7->{\B{\frac{\partial F_1}{\partial y} = 4}}}$

\vspace{0.4cm}

$\D{ \OWORK = 
\ON<8->{\B{\iint \limits_{R} (1-4)\, dxdy =\,}} 
\ON<9->{\B{-3 \iint \limits_{R} \, dxdy =\,}}
\ON<10->{\B{-3 (Area \, of \, R) = -3}}}$

\end{column}

\only<2->{\includegraphics[scale = 0.19]{Figures/10_5_Fig4.png}}
	
\end{column}
\end{columns}
