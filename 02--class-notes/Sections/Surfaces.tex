\UL{Vector Operation 3}: Cross Product

\vspace{0.3cm}

\vspace{0.3cm}

%--------EQN----------
\begin{center}
	$\VA \times \VB = \ON<2->{\B{|\VA|\,|\VB|\cos \theta}}$ \\

	$\text{or}$\\
	\vspace{0.2cm}
	$\VA \times \VB = \AAA \times \BBB = \ON<3->{\B{a_1b_1 + a_2b_2 + a_3b_3}}$
