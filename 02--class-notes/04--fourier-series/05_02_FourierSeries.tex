\UL{III. Fourier Coefficients for Even and Odd Functions}

\vspace{0.2cm}

%--------TABLE---------
%\setlength{\extrarowheight}{0.2cm}
\begin{tabular}{p{0.2cm}ll}
	\multicolumn{3}{l}{Rewriting the Fourier Coefficients ($d_n$) using Euler's Formula $\D{(e^{i\theta}=\cos\theta +i\sin\theta}) $ we observe}\\
%	$d_0$ & $\D{= 
%	\frac{1}{2L}\int \limits_{-L}^{L} f(x) \, dx}$ & \quad(No Change)
%	\\
	$d_n$ & $\D{= 
	\frac{1}{2L}\int \limits_{-L}^{L} f(x)\, e^{-i\pi nx/L} \, dx}$
	& $\D{= 
	\frac{1}{2L}\int \limits_{-L}^{L} f(x)\, \left[\,\, 
	\ON<2->{\B{\cos\left(\frac{-\pi nx}{L}\right) + i \sin\left(\frac{-\pi nx}{L}\right)\,\,}}\right] \, dx}$ \\
	&& $\D{= 
	\frac{
	\ON<3->{\B{1}}}{2L}\int \limits_{-L}^{L} f(x)\, 
	\only<5->{\US{\B{ even }}}{
	\only<5->{\UBT}{ 
	\ON<3->{\B{\,\,\cos\left(\frac{\pi nx}{L}\right)\,\,}} }}\,dx + \frac{
	\ON<4->{\B{-i}}}{2L}\int \limits_{-L}^{L} f(x)\, 
	\only<6->{\US{\B{ odd }}}{
	\only<6->{\UBT}{ 
	\ON<4->{\B{\,\,\sin\left(\frac{\pi nx}{L}\right)\,\,}} }} \, dx}$ \\
\end{tabular}
%--------------------

\vspace{0.2cm}

Therefore, if 
%--------TABLE---------
%\setlength{\extrarowheight}{0.2cm}
\begin{tabular}{p{1.4cm}p{-0.2cm}l}
	$f(x)$ is even:
	& $d_0$ & $\D{= 
	\frac{1}{2L}\int \limits_{-L}^{L} 
	\only<7->{\OS{\B{ even }}}{
	\only<7->{\OBT}{ f(x) }} \, dx}$ 
	$\D{= 
	\ON<8->{\B{\frac{1}{L}\int \limits_{0}^{L} f(x) \, dx}}}$ 
	\\
	& $d_n$ & $\D{= 
	\frac{1}{2L}\int \limits_{-L}^{L} 
	\only<10->{\OS{\B{ even }}}{
	\only<10->{\OBT}{
	\only<9->{\US{\B{ even }}}{
	\only<9->{\UBT}{ f(x) }}
	\only<9->{\US{\B{ even }}}{
	\only<9->{\UBT}{ \cos\left(\frac{\pi nx}{L}\right) }}}}
	\,dx \,+\,
	\frac{-i}{2L}\int \limits_{-L}^{L} 
	\only<12->{\OS{\B{ odd }}}{
	\only<12->{\OBT}{
	\only<11->{\US{\B{ even }}}{
	\only<11->{\UBT}{ f(x) }}
	\only<11->{\US{\B{ odd }}}{
	\only<11->{\UBT}{ \sin\left(\frac{\pi nx}{L}\right) }}}}
	\,dx}$
	= 
	$\D{
	\ON<13->{\B{\frac{1}{L}\int \limits_{0}^{L} f(x) \cos\left(\frac{\pi nx}{L}\right)	\,dx}}}$ \\
	\\
	\\
	$f(x)$ is odd:
	& $d_0$ & $\D{= 
	\frac{1}{2L}\int \limits_{-L}^{L} 
	\only<14->{\OS{\B{ odd }}}{
	\only<14->{\OBT}{ f(x) }} \, dx}$ 
	$\D{= 
	\ON<15->{\B{0}}}$ 
	\\
	& $d_n$ & $\D{= 
	\frac{1}{2L}\int \limits_{-L}^{L} 
	\only<17->{\OS{\B{ odd }}}{
	\only<17->{\OBT}{
	\only<16->{\US{\B{ odd }}}{
	\only<16->{\UBT}{ f(x) }}
	\only<16->{\US{\B{ even }}}{
	\only<16->{\UBT}{ \cos\left(\frac{\pi nx}{L}\right) }}}}
	\,dx \,+\,
	\frac{-i}{2L}\int \limits_{-L}^{L} 
	\only<19->{\OS{\B{ even }}}{
	\only<19->{\OBT}{
	\only<18->{\US{\B{ odd }}}{
	\only<18->{\UBT}{ f(x) }}
	\only<18->{\US{\B{ odd }}}{
	\only<18->{\UBT}{ \sin\left(\frac{\pi nx}{L}\right) }}}}
	\,dx}$
	= 
	$\D{
	\ON<20->{\B{\frac{-i}{L}\int \limits_{0}^{L} f(x) \sin\left(\frac{\pi nx}{L}\right)	\,dx}}}$ \\
\end{tabular}
%--------------------

\UL{IV. Purely Cosine and Sine Fourier Series}

\vspace{0.2cm}

When working with Fourier Series, the following results are often useful:
\begin{center}
{\setlength{\extrarowheight}{-0.0cm}
\begin{tabular}{|p{0.2cm}p{10.6cm}|}
	\hline
	\hline
	&\\
	(1) & If \UL{$f(t)$ is an even function}, its Fourier Series can be written purely in terms of 
	\ON<2->{\B{cosine.}}\\
	&\\
	\multicolumn{2}{|c|}{$\D{f(t) = \sum \limits_{n=-\infty}^{\infty} d_n \, e^{i\pi nt/L} = 
	\ON<3->{\B{\, d_0\,\,}} + \sum \limits_{n=1}^{\infty} 
	\ON<4->{\B{\, 
	\only<5->{\OS{\B{ a_n }}}{
	\only<5->{\OBT}{ 2d_n }} }} \,\cos\left(\frac{\pi nt}{L}\right) }$ \qquad 
	\ON<6->{\B{(cosine series)}}} \\
	&\\
	\multicolumn{2}{|c|}{$\D{
	\ON<7->{\B{a_n = 2d_n = }}
	\ON<8->{\B{\,\, 2\left[\frac{1}{L}\int \limits_{0}^{L} f(t) \cos\left(\frac{\pi nt}{L}\right) \,dt \right] = \frac{2}{L}\int \limits_{0}^{L} f(t) \cos\left(\frac{\pi nt}{L}\right) \,dt}} }$ }\\
	&\\
	(2) & If \UL{$f(t)$ is an odd function}, its Fourier Series can be written purely in terms of 
	\ON<9->{\B{sine.}}\\
	&\\
	\multicolumn{2}{|c|}{$\D{f(t) = \sum \limits_{n=-\infty}^{\infty} d_n \, e^{i\pi nt/L} = \sum \limits_{n=1}^{\infty} 
	\ON<10->{\B{\,\,
	\only<11->{\OS{\B{ b_n }}}{
	\only<11->{\OBT}{ 2id_n }} \,\,}} \,\sin\left(\frac{\pi nt}{L}\right) }$ \qquad 
	\ON<12->{\B{(sine series)}} } \\
	&\\
	\multicolumn{2}{|c|}{$\D{
	\ON<13->{\B{b_n = 2id_n = }}
	\ON<14->{\B{\,\, 2i\left[\frac{-i}{L}\int \limits_{0}^{L} f(t) \sin\left(\frac{\pi nt}{L}\right)	\,dt\right] = \frac{2}{L}\int \limits_{0}^{L} f(t) \sin\left(\frac{\pi nt}{L}\right)	\,dt }} }$}\\
	&\\
	\hline
	\hline
\end{tabular}}
\end{center}

\EX 	Find the Fourier Series of \,\, 
$f(x)=\left\{
\begin{array}{cc}
	0 & -2 \le x \le -1 \\
	1 & -1 \le x \le 1 \\
	0 & 1 \le x \le 2 \\
\end{array}
\right.
$
\quad and \quad $f(x) = f(x+4)$
 
\begin{columns}
\begin{column}{6.8cm}

Notes:

\vspace{0.2cm}

$\cdot$ \, Since $f(x) = f(x+4) \,\, \implies \,\, $ Period $= 
\ON<2->{\B{4 \,\, \implies \,\, L=2}}$

\vspace{0.2cm}

$\cdot$ \, $f(x)$ is an 
\ON<3->{\B{\,\, even \,\,}} function

$
\setlength{\extrarowheight}{0.5cm}
\begin{array}{cl}
	  \ON<6->{\B{d_0}} 
	& \ON<7->{\B{\, = \D{\frac{1}{L}\int \limits_{0}^{L} f(x)\, dx}}}
	  \ON<8->{\B{\,\, = \D{\frac{1}{2}\int \limits_{0}^{2} f(x)\, dx}}}
	  \ON<9->{\B{\,\, = \D{\frac{1}{2}\int \limits_{0}^{1} 1\, dx = \frac{1}{2}}}} \\
	  \ON<10->{\B{d_n}} 
	& \ON<11->{\B{\, = \D{\frac{1}{L}\int \limits_{0}^{L} f(x) \cos\left(\frac{n\pi x}{L}\right)\, dx} }} \\
	& \ON<12->{\B{\, = \D{\frac{1}{2}\int \limits_{0}^{2} f(x) \cos\left(\frac{n\pi x}{2}\right)\, dx} }} \\
	& \ON<13->{\B{\, = \D{\frac{1}{2}\int \limits_{0}^{1} 1 \cos\left(\frac{n\pi x}{2}\right)\, dx} }} 
	  \ON<14->{\B{\,\, = \D{\frac{1}{2}\left[\frac{2}{\pi n}\sin\left(\frac{n\pi x}{2}\right)\right]_{0}^{1}} }} 
	  \ON<15->{\B{\,\, = \D{\frac{1}{\pi n}\left(\sin\left(\frac{n\pi}{2}\right) - \sin\left(0\right)\right)} }} 
	  \ON<16->{\B{\,\, = \D{\frac{1}{\pi n}\sin\left(\frac{\pi}{2}n\right)} }} \\
\end{array}
$

$\D{\implies \quad f(x) = \sum \limits_{n=-\infty}^{\infty} d_ne^{i n\pi x/L}}$ \quad 
where \quad
$
\ON<17->{\B{d_n = \left\{
\begin{array}{cl}
	\D{\frac{1}{2}} & n=0 \\
	\D{\frac{1}{\pi n}\sin\left(\frac{\pi}{2}n\right)} & n\ne0 \\
\end{array}
\right.}}
$
