\vspace{0.2cm}

\DEF A function $f(x)$ is {\bf periodic} with {\bf period} $p$ if 
\begin{center}
	\ON<2->{\B{$f(x+p)=f(x)$ \hspace{1.0cm} \text{for all }x }}
\end{center}

\begin{center}
	\includegraphics[scale = 0.3]{Figures/11_1_Fig1.png}
\end{center}	

\EX $f(x) = \cos x$ \,\, is periodic with period \,\, 
\ON<3->{\B{$2\pi$}}
\ON<4->{\B{$\quad \implies \quad \cos(x+2\pi) = \cos(x)$}}

\begin{center}
	\only<1-4>{\includegraphics[scale = 0.3]{Figures/11_1_Fig9.png}}
	\only<5->{\includegraphics[scale = 0.3]{Figures/11_1_Fig14.png}}

\begin{frame}
%\frametitle{III. Fourier Series}
\scriptsize
\begin{columns}\begin{column}{12.0cm}

\begin{center}
%--------TABLE---------
\setlength{\extrarowheight}{0.2cm}
\begin{tabular}{|p{0.5cm}l|}
	\hline
	\hline
	\DEF & Suppose that $f(x)$ is periodic with period $2\pi$.  The {\bf Fourier series} of $f$ is \\
	\multicolumn{2}{|c|}{$\D{f(x) = 
	\ON<2->{\B{\sum \limits_{n=-\infty}^{\infty} d_n \, e^{inx}}}}$} \\
	& where $d_n$ are the complex {\bf Fourier coefficients} of $f(x)$ given by \\
	\multicolumn{2}{|c|}{$\D{d_n = 
	\ON<3->{\B{\frac{1}{2\pi}\int \limits_{-\pi}^{\pi} f(x)\, e^{-inx} \, dx}}}$} \\
	\hline
	\hline
\end{tabular}
%--------------------
\end{center}

Using Euler's formula $\D{\left(\,\,\,
\ON<4->{\B{e^{i\theta} = \cos \theta + i \sin \theta}} \,\,\, \right)}$ the fourier series can be written as:

\begin{center}
%--------TABLE---------
\setlength{\extrarowheight}{0.2cm}
\begin{tabular}{|p{0.1cm}p{2.8cm}p{3.7cm}p{3cm}p{0.1cm}|}
	\hline
	\hline
	\multicolumn{5}{|c|}{$\D{f(x) = 
	\ON<5->{\B{\, a_0 + \sum \limits_{n=1}^{\infty} \left(a_n\cos(nx) + b_n \sin(nx)\right)}}}$} \\
	\multicolumn{5}{|l|}{where $a_0$, $a_n$, and $b_n$ are the real {\bf Fourier coefficients} of $f(x)$ given by} \\
	&$\D{a_0 = 
	\ON<6->{\B{\frac{1}{2\pi}\int \limits_{-\pi}^{\pi} f(x)\, dx}}}$ &
	$\D{a_n = 
	\ON<7->{\B{\frac{1}{\pi}\int \limits_{-\pi}^{\pi} f(x) \cos(nx)\, dx}}}$ &
	$\D{b_n = 
	\ON<8->{\B{\frac{1}{\pi}\int \limits_{-\pi}^{\pi} f(x) \sin(nx)\, dx}}}$ &\\
	\hline
	\hline
\end{tabular}
%--------------------
\end{center}

\vspace{0.2cm}

The main goal for this section is to determine either $d_n$ or $a_n$ \& $b_n$. 
 
\end{column}\begin{column}{0.0cm}\end{column}\end{columns}

\begin{column}{5.6cm}
	$f(x) = \ON<2->{\B{\left\{
\begin{array}{rr}
	\R{-1} & \R{-\pi < x < 0} \\
	\G{1} & \G{0 < x < \pi} \\
\end{array}
\right.
\hspace{0.2cm} f(x+2\pi) = f(x)}}$
\end{column}
\end{columns}

\setlength{\extrarowheight}{0.2cm}
\begin{tabular}{p{0.01cm}l}
	$d_n$ & 
	\ON<3->{\B{= $\D{\frac{1}{2\pi}\int \limits_{-\pi}^{\pi} f(x)\, e^{-inx} \, dx}$ }}
	\ON<4->{\B{= $\D{\frac{1}{2\pi} \left[\R{\int \limits_{-\pi}^{0} f(x)\, e^{-inx} \, dx}\right.} $ + }}
	\ON<5->{\B{ $\D{\left.\G{\int \limits_{0}^{\pi} f(x)\, e^{-inx} \, dx}\right]}$}}
	\ON<6->{\B{= $\D{\frac{1}{2\pi} 
	     \only<7->{\OS{\B{ (1) }}}{
	     \only<7->{\OBT}{ \left[\int \limits_{-\pi}^{0} -e^{-inx} \, dx \right.}} + 
	     \only<8->{\OS{\B{ (2) }}}{
	     \only<8->{\OBT}{ \left.\int \limits_{0}^{\pi} e^{-inx} \, dx \right]}}}$}}
		\\
		\\
	$d_0$ & 
	\ON<9->{\B{= $\D{\frac{1}{2\pi}\left[\int \limits_{-\pi}^{0} -e^{0} \, dx + \int \limits_{0}^{\pi} e^{0} \, dx\right]}$}}
	\ON<10->{\B{= $\D{\frac{1}{2\pi}\left[\int \limits_{-\pi}^{0} -1 \, dx + \int \limits_{0}^{\pi} 1 \, dx\right]}$}}
	\ON<11->{\B{= $\D{ = \frac{1}{2\pi}\left[-\pi + \pi\right] = 0}$}} \\
	\\
	$(1)$ & 
	\ON<12->{\B{= $\D{\int \limits_{-\pi}^{0} -e^{-inx} \, dx}$}}
	\ON<13->{\B{= $\D{\left. \only<2->{+}\only<1>{-} \frac{1}{\only<2->{+}\only<1>{-}in}e^{-inx} \right|_{-\pi}^{0} }$}} 
	\ON<14->{\B{= $\D{\frac{1}{in}\left[ e^{0} - e^{-i\pi n} \right] }$}}
	\ON<15->{\B{= $\D{\frac{1}{in}\left[ 1 - (-1)^n \right] }$}} \\
	\\
	$(2)$ & 
	\ON<16->{\B{= $\D{\int \limits_{0}^{\pi} e^{-inx} \, dx}$}}
	\ON<17->{\B{= $\D{\left. \frac{1}{-in}e^{-inx} \right|_{0}^{\pi} }$}} 
	\ON<18->{\B{= $\D{-\frac{1}{in}\left[ e^{-i\pi n} - e^0 \right] }$}}
	\ON<19->{\B{= $\D{-\frac{1}{in}\left[(-1)^n - 1 \right] }$}} \\
\end{tabular}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}

\begin{frame}
\frametitle{}
\scriptsize
\begin{columns}\begin{column}{12.0cm}

\begin{tabular}{p{0.01cm}l}
	$d_n$ & 
	\ON<2->{\B{= $\D{\frac{1}{2\pi}\left[(1)+(2)\right]}$}}
	\ON<3->{\B{= $\D{\frac{1}{2\pi}\left[\frac{1}{in}\left[ 1 - (-1)^n \right] + \frac{1}{in}\left[(-1)^n - 1 \right]\right]}$}}
	\ON<4->{\B{= $\D{\frac{1}{2i\pi n}\left[ 2 - 2(-1)^n \right]}$}}
	\ON<5->{\B{= $\D{\frac{1 - (-1)^n}{i\pi n}}$}}
\end{tabular}

\vspace{0.6cm}

\ON<6->{\B{Note:}}  \quad 
$
\ON<7->{\B{d_n = \left\{
\begin{array}{cl}
	\ON<8->{\B{0}} & 
	\ON<9->{\B{n \text{ even}}}\\
	\ON<10->{\B{\D{\frac{2}{i \pi n}}}} & 
	\ON<11->{\B{n \text{ odd}}}\\
\end{array}
\right.}}
$

\vspace{0.6cm}

Therefore,

\vspace{0.5cm}

$\D{
f(x) = \sum \limits_{n=-\infty}^{\infty} d_n \, e^{inx} = \,
\ON<12->{\B{\sum \limits_{n=\pm odd} \frac{2}{i\pi n} \, e^{inx}}}
}$

\vspace{4.2cm}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}
