{II. Gradient}
\scriptsize
\begin{center}
{\setlength{\extrarowheight}{-0.2cm}
\begin{tabular}{|p{0.4cm}p{10.6cm}|}
	\hline
	\hline
	&\\
	\DEF & Let $D_f$ be the domain of the scalar function $f(x,y)$ and $D_g$ be the domain of the scalar function $g(x,y,z)$.  If $f$ is differentiable on $D_f$ and $g$ is differentiable on $D_g$ then the {\bf gradient} of $f(x,y)$ and $g(x,y,z)$ is given by \\
	&\\
	\multicolumn{2}{|c|}{$\D{\nabla f = 
	\ON<1>{\B{\left[\PFPX,\PFPY\right] =\,}} 
	\ON<1>{\B{\left[f_x,f_y\right] =\,}} 
	\ON<1>{\B{f_x \VI + f_y \VJ}}}$} \\
	& \\
	\multicolumn{2}{|c|}{$\D{\nabla g = 
	\ON<1>{\B{\left[\PGPX,\PGPY,\PGPZ\right] =\,}} 
	\ON<1>{\B{\left[g_x,g_y,g_z\right] =\,}} 
	\ON<1>{\B{g_x \VI + g_y \VJ + g_z \VK}}}$} \\
	& \\
	\hline
	\hline
\end{tabular}}
\end{center}
\EX $f(x,y) = xy + y$. \quad Find $\GF$

\vspace{0.2cm}

$\D{\ON<1>{\B{\PFPX =\,}} 
\ON<1>{\B{y}}}$ \quad 
\ON<1>{\B{and}} \quad $\D{
\ON<1>{\B{\PFPY =\,}} 
\ON<1>{\B{x + 1}}}$ \quad 
\ON<1>{\B{$\implies$}} \quad 
\ON<1>{\B{$\D{\GF = [y,x+1]}$}}

\vspace{0.2cm}

\EX $f(x,y,z) = z\ln(xy^2 + y)$. \quad Find $\GF$ at the point $P(0,1,2)$

\vspace{0.1cm}

$\D{\ON<1>{\B{\PFPX =\,}} 
\ON<1>{\B{z\frac{1}{xy^2 + y}\cdot \PPX(xy^2 + y) = \,}} 
\ON<1>{\B{z\frac{1}{xy^2 + y}\cdot y^2 =\,}} 
\ON<1>{\B{\frac{zy^2}{xy^2 + y}}}}$

\vspace{0.1cm}

$\D{\ON<1>{\B{\PFPY = \,}}
\ON<1>{\B{z\frac{1}{xy^2 + y}\cdot \PPY(xy^2 + y) = \,}}
\ON<1>{\B{z\frac{1}{xy^2 + y}\cdot (2xy+1) =\,}}
\ON<1>{\B{\frac{z(2xy+1)}{xy^2 + y}}}}$

\vspace{0.1cm}

$\D{\ON<1>{\B{\PFPZ = \,}}
\ON<1>{\B{\ln(xy^2 + y) }}}$

\vspace{0.1cm}

$\D{\ON<1>{\B{ \implies \, \GF(P) = \left[\PFPX(P),\PFPY(P),\PFPZ(P)\right] = \,}} 
\ON<1>{\B{\left[ \,
\ON<1>{\B{\,\frac{(2)(1)^2}{(0)(1)^2 + (1)}\, }},
\ON<1>{\B{\,\frac{(2)(2(0)(1)+1)}{(0)(1)^2 + (1)}\,}},
\ON<1>{\B{\,\ln((0)(1)^2 + (1))}}\right] = \,}}
\ON<1>{\B{[2,2,0]}}}$

\vspace{0.2cm}

\scriptsize

Prove the following two properties of gradients: \qquad Recall:  $\D{\nabla f = \left[\PFPX,\PFPY,\PFPZ \right]}$

\quad 1. $\D{\quad \nabla(f+g) = \GF + \GG}$
%--------EQN----------
\begin{align*}
Proof: \quad \nabla(f+g)
&\,=
\ON<1>{\B{ \left[\PPX(f+g)\,,\,\PPY(f+g)\,,\,\PPZ(f+g)\right] }} \hspace{0.3cm} 
\ON<1>{\B{(Defn \, of \, Gradient)}} \hspace{1.9cm}\\
&\,
\ON<1>{\B{\, = \left[\PFPX+\PGPX\,,\,\PFPY+\PGPY\,,\,\PFPZ+\PGPZ\right]}} \hspace{0.3cm} 
\ON<1>{\B{(Addition \, Rule \times3)}} \\
&\,
\ON<1>{\B{\, = \left[\PFPX\,,\,\PFPY\,,\,\PFPZ\right] + \left[\PGPX\,,\,\PGPY\,,\,\PGPZ\right]}} \hspace{0.3cm} 
\ON<1>{\B{(Vector \, Addition)}}\\
&\,
\ON<1>{\B{\, = \GF + \GG}} \hspace{0.3cm} 
\ON<1>{\B{(Defn \, of \, Gradient \times 2)}}
\end{align*}
%----------------------

\vspace{0.4cm}

\quad 2. $\D{\quad \nabla(f \cdot g) = \GF \cdot g + f \cdot \GG}$
%--------EQN----------
\begin{align*}
Proof: \quad \nabla(fg)
&\,=
\ON<1>{\left[\B{(f \cdot g)_x}\,,\,\R{(f \cdot g)_y}\,,\,\G{(f \cdot g)_z}\right] }\hspace{0.3cm} 
\ON<1>{\B{(Defn \, of \, Gradient)}} \hspace{3.1cm}\\
&\,
\ON<1>{= \left[\B{f_x \cdot g + f  \cdot g_x}\,,\,\R{f_y \cdot g + f \cdot g_y}\,,\,\G{f_z \cdot g + f \cdot g_z}\right]}\hspace{0.3cm} 
\ON<1>{\B{(Product \, Rule \times3)}}\\
&\,
\ON<1>{= \left[\B{f_x \cdot g}\,,\,\R{f_y\cdot g}\,,\,\G{f_z\cdot g}\right] + \left[\B{f \cdot g_x}\,,\,\R{f \cdot g_y}\,,\,\G{f \cdot g_z}\right]} \hspace{0.3cm} 
\ON<1>{\B{(Vector \, Addition)}}\\
&\,
\ON<1>{= \left[\B{f_x} \,,\,\R{f_y}\,,\,\G{f_z}\right] \cdot g + f \cdot \left[\B{g_x}\,,\,\R{g_y}\,,\,\G{g_z}\right]} \hspace{0.3cm} 
\ON<1>{\B{(Scalar \cdot Vector)}}\\
&\,
\ON<1>{\B{= \GF \cdot g + f \cdot \GG}} \hspace{0.3cm} 
\ON<1>{\B{(Defn \, of \, Gradient \times 2)}}
\\
\end{align*}
%----------------------

\vspace{0.8cm}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}

\scriptsize	

\EX Is there a direction {\bf u} in which the rate of change of the temperature function \,\, $T(x,y) = x^2 - 3xy + 4y^2$ \,\, at \,\, $P(1, 2)$ \,\,  equals \,\,  $14^{\circ} C/ft$? Give reasons for your answer.
	
	\vspace{0.2cm}
	
ANSWER: \ON<1>{\B{No}}

\vspace{0.2cm}

REASON: 

\vspace{0.2cm}

$\D{
\ON<1>{\B{\nabla T = }}
\ON<1>{\B{[2x-3y,-3x+8y]}}
\ON<1>{\B{ \quad \implies \quad \nabla T(P) = [2(1)-3(2),-3(1)+8(2)]}}
\ON<1>{\B{\,= [-4,13]}}
}$

\vspace{0.2cm}

$\D{
\ON<1>{\B{ \hspace{2.6cm} \qquad \implies \quad |\nabla T(P)| = |[-4,13]| = \sqrt{(-4)^2 + (13)^2} = \sqrt{185} \approx 13.6}}
}$

\vspace{0.5cm}

$\D{
\ON<1>{\B{\implies \quad \text{Max rate of change at $P$ is:}}}
\ON<1>{\B{\sqrt{185}}}}$

\vspace{0.5cm}

$\D{
\ON<1>{\B{\implies \quad \text{Min rate of change at $P$ is:}}}
\ON<1>{\B{-\sqrt{185}}}}$

\vspace{0.5cm}

\ON<1>{\B{Therefore, since 14 does not lie within $\left[\sqrt{185}, -\sqrt{185}\right]$, the rate of change of the temperature can never be equal to \,\, $14^{\circ} C/ft$}}


\vspace{4.2cm}
