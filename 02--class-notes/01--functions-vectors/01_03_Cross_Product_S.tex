{I. Definition}

\vspace{0.2cm}

\scriptsize
	
\UL{Vector Operation 3}: Cross Product

\vspace{0.3cm}

Let \,\, $\VA = \AAA$, \, $\VB = \BBB$, and \, {\bf n} is a unit vector perpendicular to both $\VA$ and $\VB$

\vspace{0.3cm}

Then 
%--------EQN----------
\begin{center}
	$\VA \times \VB = (|\VA|\,|\VB|\sin \theta) \, {\bf n}$ \\
	\vspace{0.2cm}
	$\text{and so}$\\
	\vspace{0.2cm}
	$|\VA \times \VB| = \ON<0>{\B{|\VA|\,|\VB|\sin \theta}}$
\end{center}
%----------------------

\vspace{0.2cm}

\UL{Notes}:

\vspace{0.1cm}

\OBUL $\VA,\VB \in \RE^3 \quad \implies \quad 
\ON<0>{\B{\VA \times \VB \in \RE^3 \,\, (vector)}}$ and \,\,
\ON<0>{\B{$\VA, \VB \perp \VA \times \VB$}}\\

\vspace{0.2cm}

\OBUL $\VA \times \VB 
\ON<0>{\B{\,\,\ne\,\,}} \VB \times \VA \quad 
\ON<0>{\B{(not \, commutative)}}$ 

\vspace{0.2cm}

\OBUL A way to remember how to calculate a vector product is to use determinants:
%--------EQN----------
\begin{align*}
	\VA \times \VB = 
\ON<0>{\B{
\begin{array}{|ccc|}
	{\bf i}&{\bf j}&{\bf k} \\
	a_1 & a_2 & a_3 \\
	b_1 & b_2 & b_3 \\
\end{array}
}}
& 
\ON<0>{\B{\,\,=
\begin{array}{|cc|}
	\ON<0>{\B{a_2}} & \ON<0>{\B{a_3}} \\
	\ON<0>{\B{b_2}} & \ON<0>{\B{b_3}} \\
\end{array}
}}
\,\,\ON<0>{\B{{\bf i} -}}
\ON<0>{\B{\,\,
\begin{array}{|cc|}
	\ON<0>{\B{a_1}} & \ON<0>{\B{a_3}} \\
	\ON<0>{\B{b_1}} & \ON<0>{\B{b_3}} \\
\end{array}
}}
\,\, \ON<0>{\B{{\bf j} + }}
\ON<0>{\B{\,\,
\begin{array}{|cc|}
	\ON<0>{\B{a_1}} & \ON<0>{\B{a_2}} \\
	\ON<0>{\B{b_1}} & \ON<0>{\B{b_2}} \\
\end{array}
}}
\,\,\ON<0>{\B{{\bf k}}} \\
& \ON<0>{\B{= (a_2b_3 - a_3b_2){\bf i} - (a_1b_3 - a_3b_1){\bf j} + (a_1b_2 - a_2b_1){\bf k}}} \\
& \ON<0>{\B{= [a_2b_3 - a_3b_2, \,\, a_3b_1 - a_1b_3, \,\, a_1b_2 - a_2b_1]}}
\end{align*}
%----------------------

{II. Additional Concepts}

\vspace{0.2cm}

\scriptsize

\UL{Geometric Interpretation 1}:
\begin{columns}
\begin{column}{3.3cm}
\begin{center}
	\includegraphics[scale = 0.18]{\FIGSDIR/9_3_Fig4.png}
\end{center}
	
\end{column}
\begin{column}{7.3cm}
\begin{center}
	Dot Product:
	
	\vspace{0.1cm}
	
	$\VA \cdot \VB = 
	\ON<0>{\text{(\B{component of $\VA$ along $\VB$})}}$
	\ON<0>{(\textcolor{Melon}{length of $\VB$})} \\
	\ON<0>{$=(\B{|a|\cos\theta})\,(\textcolor{Melon}{|b|})\,$}
	\ON<0>{$=\B{|a|}\,\textcolor{Melon}{|b|}\,\B{\cos\theta}$} \hspace{0.9cm}
				  
	\vspace{0.2cm}
	
	Cross Product:
	
	\vspace{0.1cm}
	
	$|\VA \times \VB| = 
	\ON<0>{\text{(\B{component of $\VA$ perpendicular to $\VB$})}}$
	\ON<0>{(\textcolor{Melon}{length of $\VB$})} \\
	\ON<0>{$ = (\B{|a|\,\sin\theta})\,(\textcolor{Melon}{|b|})\,$}
	\ON<0>{$ = \B{|a|}\,\textcolor{Melon}{|b|}\,\B{\sin\theta} \hspace{2.14cm}$} 
	
\end{center}

\vspace{0.2cm}

\end{column}
\end{columns}

\vspace{0.2cm}

\UL{Properties}: \quad Let $\VA, \VB, \VC \in \RE^3$ and $d \in \RE \,\,(a \, scalar)$

\vspace{0.15cm}

\OBUL $d (\VA \times \VB) = 
\ON<0>{\B{d \VA \times \VB = \VA \times d\VB}}$

\vspace{0.15cm}

\OBUL $\VA \times (\VB + \VC) = 
\ON<0>{\B{\VA \times \VB + \VA \times \VC}} $

\vspace{0.15cm}

\OBUL $\VA \times \VB = 
\ON<0>{\B{-(\VB \times \VA)}}$
