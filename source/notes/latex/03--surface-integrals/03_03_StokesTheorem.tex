\UL{I. Stokes' Theorem}
\scriptsize

\vspace{0.2cm}

{\setlength{\extrarowheight}{-0.2cm}
\begin{tabular}{|p{0.4cm}p{10.6cm}|}
	\hline
	\hline
	&\\
	\THM & Let $S$ be a surface with boundary $C$ and ${\bf F}(x,y,z)$ be a vector-valued function defined on $S$.  Then\\
	&\\
	\multicolumn{2}{|c|}{
	$\D{\oint \limits_{C} {\bf F} \cdot d{\bf r} = 
	\ON<1>{\B{\iint \limits_{S}(\text{curl}\,{\bf F}) \cdot {\bf n} \, dA = \,}}
	\ON<1>{\B{\iint \limits_{S} \nabla \times {\bf F} \cdot {\bf n} \, dA}}}$}\\
	&\\
	\hline
	\hline
\end{tabular}}
\begin{center}
	\includegraphics[scale = 0.2]{\FIGSDIR/10_9_Fig1.png} \hspace{1.0cm} \includegraphics[scale = 0.2]{\FIGSDIR/10_9_Fig2.png}
\end{center}
Note that Green's Theorem is a special case of Stokes' theorem when $S$ is a plane region $R$. In this case, $\, {\bf n}$ is just \,\, \ON<1>{\B{${\bf k}=[0,0,1]$}} \,\, and
\begin{center}
	$\D{\OWORK = 
	\ON<1>{\B{\iint \limits_{S}(\text{curl}\,{\bf F}) \cdot {\bf n} \, dA =\,}}
	\ON<1>{\B{\GINT = \,}}
	\ON<1>{\B{\GREENSA}}}$
\end{center}

\end{column}\begin{column}{0.0cm}\end{column}\end{columns}
\end{frame}

\EX ${\bf F} = [-5y,4x,z]$, and $C$ is the circle $x^2 + y^2 = 16$, $z=4$. \quad Find $\D{\OWORK}$

\vspace{0.4cm}

Note:
$\D{
\text{curl}\, {\bf F} =
\ON<1>{\B{\, [0,0,9]}}}$ \quad and $\D{{\bf n} = 
\ON<1>{\B{\,[0,0,1]}}}$

\vspace{0.4cm}

So 
\ON<1>{\B{ by Stokes' Theorem and the fact $S$ is a circle}}

\vspace{0.4cm}

$\D{\OWORK = 
\ON<1>{\B{\,\STOKECURL}}
\ON<1>{\B{\, = \iint \limits_{S} 9\, dA}}
\ON<1>{\B{\, = 9 \cdot Area(S)}}
\ON<1>{\B{\, = 9 (\pi \cdot 4^2) = 144 \pi}}}$

\vspace{9.2cm}
