\stepcounter{Prob}
{\bf Problem \arabic{Prob}:} Let {\bf  F } be any nonconstant vector field of the form \(\mathbf{F} = f \left( x \right)
\mathbf{i} + g \left( y \right) \mathbf{j} + h \left( z \right) \mathbf{k}\) and let
{\bf  G } be any nonconservative vector field of the form \(\mathbf{F} = f \left( y, z \right)
\mathbf{i} + g \left( x, z \right) \mathbf{j} + h \left( x, y \right) \mathbf{k}\).
Indicate whether the following statements are true or false by placing "T" or "F" to
the left of the statement. \leavevmode\\\relax 

\par\begin{enumerate}
\item[\mbox{\parbox[t]{3ex}{\hrulefill}}1.] {\bf  G } is irrotational
\item[\mbox{\parbox[t]{3ex}{\hrulefill}}2.] {\bf  F } is irrotational
\item[\mbox{\parbox[t]{3ex}{\hrulefill}}3.] {\bf  G } is incompressible
\item[\mbox{\parbox[t]{3ex}{\hrulefill}}4.] {\bf  F } is incompressible
\end{enumerate}
