\UL{Complex Numbers}:

Let $z = re^{\theta i} = r(\cos \theta + i\sin \theta)$

\qquad \, $\D{z^n = r^ne^{n\theta i} = r^n\left[\cos (n\theta) + i\sin (n\theta)\right]}$

\qquad \, $\D{z^{1/n} = r^{1/n}e^{\frac{\theta+2\pi k}{n}} = r^{1/n}\left[\cos \left(\frac{\theta+2\pi k}{n}\right) + i\sin \left(\frac{\theta+2\pi k}{n}\right)\right]}$ \quad for \,\,$k=0,1,2,...,n-1$

\vspace{0.1cm}

\UL{Fourier Series}:	
	 
The Fourier series of a function, \,$f(x)$, \,with period, $2L$, is given by
\begin{center}
	$\D{f(x) = a_0 + \sum \limits_{n=1}^{\infty} \left[a_n\cos\left(\frac{n\pi x}{L}\right) + b_n \sin\left(\frac{n\pi x}{L}\right)\right]}$
\end{center}
where
\begin{center}
	$\D{a_0 = \frac{1}{2L}\int \limits_{-L}^{L} f(x)\, dx}$, \qquad 
	$\D{a_n = \frac{1}{L}\int \limits_{-L}^{L} f(x) \cos\left(\frac{n\pi x}{L}\right)\, dx}$, \qquad
	$\D{b_n = \frac{1}{L}\int \limits_{-L}^{L} f(x) \sin\left(\frac{n\pi x}{L}\right)\, dx}$
\end{center}
