\UL{Answers to Exercises}

\vspace{0.2cm}

1. \,\, $\D{2+2i = 8^{1/2}e^{i(\pi/4)} \,\, \implies \,\, \left(2+2i\right)^6 = \left(8^{1/2}e^{i(\pi/4)}\right)^6 = 8^{6/2}e^{i(6\pi/4)} = 8^{3}e^{i(3\pi/2)}}$

\vspace{0.2cm}

2. \,\, $\D{2+2i = 8^{1/2}e^{i(\pi/4)} \,\, \implies \,\, \left(2+2i\right)^{1/3} = \left(8^{1/2}e^{i(\pi/4)}\right)^{1/3} = 8^{1/6}e^{i(\pi/12+2\pi k/3)} \quad k=0,1,2}$

\vspace{0.1cm}

\hspace{1.0cm} $k=0: \quad 8^{1/6}e^{i(\pi/12)}$

\hspace{1.0cm} $k=1: \quad 8^{1/6}e^{i(9\pi/12)}$

\hspace{1.0cm} $k=2: \quad 8^{1/6}e^{i(17\pi/12)}$

\vspace{0.2cm}

3. \,\, $\D{4(1-\sqrt{3}i) = 8e^{i(5\pi/3)} \,\, or \,\, 8e^{i(-\pi/3)} \,\, \implies \,\, 4\left((1-\sqrt{3}i)\right)^7 = 2^2\left(2e^{i(-\pi/3)}\right)^7 = 2^{9}e^{i(-7\pi/3)}}$

\vspace{0.2cm}

4. \,\, $\D{4(1-\sqrt{3}i) = 8e^{i(5\pi/3)} \,\, or \,\, 8e^{i(-\pi/3)} \,\, \implies \,\, 4\left((1-\sqrt{3}i)\right)^{1/4} = 2^2\left(2e^{i(-\pi/3)}\right)^{1/4} = 2^{9/4}e^{i(-\pi/12+2\pi k/4)}}$ 

\hspace{15.5cm} $\D{k=0,1,2,3}$

\hspace{1.0cm} $k=0: \quad 2^{9/4}e^{i(-\pi/12)}$

\hspace{1.0cm} $k=1: \quad 2^{9/4}e^{i(5\pi/12)}$

\hspace{1.0cm} $k=2: \quad 2^{9/4}e^{i(11\pi/12)}$

\hspace{1.0cm} $k=3: \quad 2^{9/4}e^{i(17\pi/12)}$

\vspace{0.2cm}

5. \,\, $\D{-2+2i = 8^{1/2}e^{i(3\pi/4)} \,\, \text{and} \,\, 3i = 3e^{i(\pi/2)} }$

$\D{ \,\, \implies \,\, \left(\frac{-2+2i}{3i}\right)^{1/2} = \left(\frac{8^{1/2}e^{i(3\pi/4)}}{3e^{i(\pi/2)}}\right)^{1/2} = \left(\frac{8^{1/2}}{3}e^{i(\pi/4)}\right)^{1/2} = \frac{8^{1/4}}{3^{1/2}}e^{i(\pi/8+2\pi k/2)} \quad k=0,1}$ 

\vspace{0.1cm}

\hspace{1.0cm} $\D{k=0: \quad \frac{8^{1/4}}{3^{1/2}}e^{i(\pi/8)}}$

\hspace{1.0cm} $\D{k=1: \quad \frac{8^{1/4}}{3^{1/2}}e^{i(9\pi/8)}}$

\vspace{0.2cm}

6. \,\, $\D{z^{1/6}-2-2i = 0 \,\, \implies \,\, z = (2+2i)^6 = 8^{3}e^{i(3\pi/2)} \quad (from \,\, 1.)}$

\vspace{0.2cm}

7. \,\, $\D{z^3 + 5i = 2+7i \,\, \implies \,\, z = (2+2i)^{1/3} = 8^{1/6}e^{i(\pi/12+2\pi k/3)} \quad k=0,1,2 \quad (from \,\, 2.)}$

\vspace{0.2cm}

8. \,\, $\D{3iz^2 + 2 = 2i \,\, \implies \,\, z = \left(\frac{-2+2i}{3i}\right)^{1/2} = \frac{8^{1/4}}{3^{1/2}}e^{i(\pi/8+2\pi k/2)} \quad k=0,1 \quad (from \,\, 5.)}$
