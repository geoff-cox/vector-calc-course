{\bf Topics:}

\vspace{0.2cm}

%-------LIST---------
\begin{itemize}
	\item[$\Box$] Difference between Average and Instantaneous Rates of Change.
	\item[$\Box$] Approximating the Derivative.
	\item[$\Box$] Given a graph of data points, answer questions relating to rates of change.
	\item[$\Box$] Given $f(x)$, find $f(x+\Delta x)$.
	\item[$\Box$] Given $f(x)$, find $f'(x)$.
	\item[$\Box$] Given $f(x)$ and $x=c$, find the tangent line of $f(x)$ at $c$.
	\item[$\Box$] Use the tangent line of $f(x)$ to approximate $f(x)$.
	\item[$\Box$] Find derivatives using the derivative rules 
		%-------LIST---------
		\begin{itemize}
			\item[$\succ$] product rule
			\item[$\succ$] quotient rule
			\item[$\succ$] chain rule
			\item[$\succ$] implicit differentiation
		\end{itemize}
		%-------------------
	\item[$\Box$] Given $f(x)$, find the critical points of $f(x)$.
	\item[$\Box$] Given $f(x)$, find the intervals of increase and decrease of $f(x)$.
	\item[$\Box$] Given $f(x)$, find the intervals of concavity of $f(x)$.
	\item[$\Box$] Given $f(x)$, find the local maximum and minimum of $f(x)$.
	\item[$\Box$] Given $f(x)$, find the absolute maximum and minimum of $f(x)$.
\end{itemize}
